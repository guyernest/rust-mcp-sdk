```rust
use serde::Deserialize;
use std::collections::HashMap;
use std::path::Path;
use std::time::Duration;
use crate::loadtest::error::LoadTestError;

#[derive(Debug, Deserialize, Clone)]
pub struct LoadTestConfig {
    pub settings: Settings,
    pub scenario: Vec<ScenarioStep>,
}

#[derive(Debug, Deserialize, Clone)]
pub struct Settings {
    pub virtual_users: u32,
    pub duration_secs: u64,
    pub timeout_ms: u64,
    #[serde(default = "default_expected_interval")]
    pub expected_interval_ms: u64,
}

fn default_expected_interval() -> u64 { 100 }

#[derive(Debug, Deserialize, Clone)]
#[serde(tag = "type")]
pub enum ScenarioStep {
    #[serde(rename = "tools/call")]
    ToolCall {
        weight: u32,
        tool: String,
        #[serde(default)]
        arguments: serde_json::Value,
    },
    #[serde(rename = "resources/read")]
    ResourceRead {
        weight: u32,
        uri: String,
    },
    #[serde(rename = "prompts/get")]
    PromptGet {
        weight: u32,
        prompt: String,
        #[serde(default)]
        arguments: HashMap<String, String>,
    },
}
```

- `LoadTestConfig::load(path: &Path) -> Result<Self, LoadTestError>` — reads file, parses TOML, calls validate, returns config.
- `LoadTestConfig::validate(&self) -> Result<(), LoadTestError>` — checks scenario not empty, total weight > 0.
- `LoadTestConfig::from_toml(content: &str) -> Result<Self, LoadTestError>` — parse + validate (used by tests and by load).
- `Settings::timeout_as_duration(&self) -> Duration` — returns `Duration::from_millis(self.timeout_ms)`.
- `ScenarioStep::weight(&self) -> u32` — returns weight regardless of variant.

**IMPORTANT per user decision:** Target server URL is NOT in the config. It comes from --url CLI flag only. Do NOT add a url field to Settings.

Run tests — they MUST all pass. Commit.
