<objective>
Add OAuth and API key authentication support to `cargo pmcp loadtest run` using the HttpMiddlewareChain pattern (same pattern as StreamableHttpTransport and ServerTester).

Purpose: Users need to load-test OAuth-protected MCP servers. Currently `loadtest run` creates unauthenticated connections. This plan adds the same auth flags as `cargo pmcp test` and uses the middleware chain to transparently inject auth headers into every VU HTTP request.

Output: Compiling codebase where `cargo pmcp loadtest run --oauth-client-id X` acquires a token at startup, creates an OAuthClientMiddleware chain, and all VU requests have `Authorization: Bearer <token>` injected by the middleware.
</objective>

- In `send_request`, BEFORE building the reqwest request, if middleware chain is present:
  1. Serialize the body to bytes: `let body_bytes = serde_json::to_vec(body).map_err(|e| McpError::Connection { message: format!("JSON serialize error: {e}") })?;`
  2. Create an `HttpRequest`: `let mut http_req = pmcp::client::http_middleware::HttpRequest::new("POST".to_string(), self.base_url.clone(), body_bytes);`
  3. Copy existing headers to http_req: Content-Type, Accept, session header if present
  4. Create context: `let context = pmcp::client::http_middleware::HttpMiddlewareContext::new(self.base_url.clone(), "POST".to_string());`
  5. Run middleware: `chain.process_request(&mut http_req, &context).await.map_err(|e| McpError::Connection { message: format!("Middleware error: {e}") })?;`
  6. After middleware runs, build the reqwest request using the modified headers from `http_req.headers` instead of the original headers. Iterate `http_req.headers` and add each to the reqwest RequestBuilder.

  **IMPORTANT:** The middleware approach means headers added by OAuthClientMiddleware (Authorization: Bearer) are transparently injected. McpClient has ZERO auth-specific logic -- it just runs the middleware chain. This is the same pattern used by StreamableHttpTransport in `build_request_with_middleware()`.

  The full `send_request` flow when middleware is present:
  ```rust
  async fn send_request(&mut self, body: &Value) -> Result<(HeaderMap, Vec<u8>), McpError> {
      if let Some(ref chain) = self.http_middleware_chain {
          // Middleware path: build HttpRequest, run chain, then build reqwest from modified headers
          use pmcp::client::http_middleware::{HttpMiddlewareContext, HttpRequest as MwRequest};

          let body_bytes = serde_json::to_vec(body)
              .map_err(|e| McpError::Connection { message: format!("JSON serialize: {e}") })?;
          let mut http_req = MwRequest::new("POST".to_string(), self.base_url.clone(), body_bytes);
          http_req.add_header("Content-Type", "application/json");
          http_req.add_header("Accept", "application/json, text/event-stream");
          if let Some(ref sid) = self.session_id {
              http_req.add_header("mcp-session-id", sid);
          }

          let context = HttpMiddlewareContext::new(self.base_url.clone(), "POST".to_string());
          chain.process_request(&mut http_req, &context).await
              .map_err(|e| McpError::Connection { message: format!("Middleware error: {e}") })?;

          // Build reqwest request from middleware-modified headers
          let mut request = self.http.post(&self.base_url).timeout(self.request_timeout);
          for (key, value) in &http_req.headers {
              request = request.header(key, value);
          }
          request = request.body(http_req.body);

          let response = request.send().await.map_err(|e| McpError::classify_reqwest(&e))?;
          // ... same status/headers/bytes handling as non-middleware path
      } else {
          // Original path (no middleware) -- unchanged
      }
  }
  ```

Create helper function `resolve_auth_middleware` that mirrors `create_oauth_middleware` from `crates/mcp-tester/src/main.rs`:

```rust
use pmcp::client::http_middleware::HttpMiddlewareChain;
use std::sync::Arc;

async fn resolve_auth_middleware(
    mcp_server_url: &str,
    api_key: Option<String>,
    oauth_client_id: Option<String>,
    oauth_issuer: Option<String>,
    oauth_scopes: Option<Vec<String>>,
    oauth_no_cache: bool,
    oauth_redirect_port: u16,
) -> anyhow::Result<Option<Arc<HttpMiddlewareChain>>> {
    // API key takes precedence (simpler, no flow needed)
    // For API key: create a simple middleware chain with a BearerToken
    if let Some(key) = api_key {
        use pmcp::client::oauth_middleware::{BearerToken, OAuthClientMiddleware};
        eprintln!("Using API key authentication");
        let bearer_token = BearerToken::new(key);
        let middleware = OAuthClientMiddleware::new(bearer_token);
        let mut chain = HttpMiddlewareChain::new();
        chain.add(Arc::new(middleware));
        return Ok(Some(Arc::new(chain)));
    }

    // OAuth flow if client_id is provided -- uses OAuthHelper.create_middleware_chain()
    if let Some(client_id) = oauth_client_id {
        use mcp_tester::{OAuthConfig, OAuthHelper};
        use mcp_tester::oauth::default_cache_path;

        let scopes = oauth_scopes.unwrap_or_else(|| vec!["openid".to_string()]);
        let cache_file = if oauth_no_cache { None } else { Some(default_cache_path()) };

        let config = OAuthConfig {
            issuer: oauth_issuer,
            mcp_server_url: Some(mcp_server_url.to_string()),
            client_id,
            scopes,
            cache_file,
            redirect_port: oauth_redirect_port,
        };

        let helper = OAuthHelper::new(config)?;
        // create_middleware_chain handles: get_access_token, create BearerToken,
        // wrap in OAuthClientMiddleware, build chain -- all in one call
        let chain = helper.create_middleware_chain().await?;
        return Ok(Some(chain));
    }

    // Warn if issuer provided without client_id
    if oauth_issuer.is_some() {
        eprintln!("Warning: --oauth-issuer provided but --oauth-client-id missing. OAuth disabled.");
    }

    Ok(None)
}
```
