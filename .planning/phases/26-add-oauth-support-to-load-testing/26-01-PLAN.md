<objective>
Add OAuth and API key authentication support to `cargo pmcp loadtest run` by threading auth headers from CLI flags through the engine, VU loop, and McpClient.

Purpose: Users need to load-test OAuth-protected MCP servers. Currently `loadtest run` creates unauthenticated connections, so any OAuth-protected server rejects VU requests. This plan adds the same auth flags as `cargo pmcp test` and injects auth headers into every VU HTTP request.

Output: Compiling codebase where `cargo pmcp loadtest run --oauth-client-id X` acquires a token at startup and all VU requests include `Authorization: Bearer <token>`.
</objective>

```rust
async fn resolve_auth_header(
    mcp_server_url: &str,
    api_key: Option<String>,
    oauth_client_id: Option<String>,
    oauth_issuer: Option<String>,
    oauth_scopes: Option<Vec<String>>,
    oauth_no_cache: bool,
    oauth_redirect_port: u16,
) -> anyhow::Result<Option<String>> {
    // API key takes precedence (simpler, no flow needed)
    if let Some(key) = api_key {
        eprintln!("Using API key authentication");
        return Ok(Some(format!("Bearer {}", key)));
    }

    // OAuth flow if client_id is provided
    if let Some(client_id) = oauth_client_id {
        use mcp_tester::{OAuthConfig, OAuthHelper};
        use mcp_tester::oauth::default_cache_path;

        let scopes = oauth_scopes.unwrap_or_else(|| vec!["openid".to_string()]);
        let cache_file = if oauth_no_cache { None } else { Some(default_cache_path()) };

        let config = OAuthConfig {
            issuer: oauth_issuer,
            mcp_server_url: Some(mcp_server_url.to_string()),
            client_id,
            scopes,
            cache_file,
            redirect_port: oauth_redirect_port,
        };

        let helper = OAuthHelper::new(config)?;
        let token = helper.get_access_token().await?;
        return Ok(Some(format!("Bearer {}", token)));
    }

    // Warn if issuer provided without client_id
    if oauth_issuer.is_some() {
        eprintln!("Warning: --oauth-issuer provided but --oauth-client-id missing. OAuth disabled.");
    }

    Ok(None)
}
```

Pass auth_header to engine builder: `engine = engine.with_auth_header(auth_header);` (after the `with_no_color` call).
