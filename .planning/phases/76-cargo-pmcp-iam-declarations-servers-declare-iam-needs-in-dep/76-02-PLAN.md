```rust
/// IAM declarations for the deployed Lambda's execution role.
///
/// Translated to `mcpFunction.addToRolePolicy(...)` calls in the generated CDK
/// stack.ts. Empty default preserves byte-identity for servers that do not
/// declare any IAM (backward compatibility — CONTEXT.md D-05).
///
/// See `/Users/guy/Development/mcp/sdk/pmcp-run/docs/CLI_IAM_CHANGE_REQUEST.md`
/// for the authoritative schema.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct IamConfig {
    /// DynamoDB table permissions (sugar form — see TablePermission).
    #[serde(default)]
    pub tables: Vec<TablePermission>,

    /// S3 bucket permissions (sugar form — object-level ARNs only).
    #[serde(default)]
    pub buckets: Vec<BucketPermission>,

    /// Raw IAM policy statements (passthrough after validation).
    #[serde(default)]
    pub statements: Vec<IamStatement>,
}

impl Default for IamConfig {
    fn default() -> Self {
        Self {
            tables: vec![],
            buckets: vec![],
            statements: vec![],
        }
    }
}

impl IamConfig {
    /// Returns true when no IAM declarations are present.
    ///
    /// Used by `DeployConfig`'s `#[serde(skip_serializing_if = "IamConfig::is_empty")]`
    /// to preserve byte-identity for configs without an `[iam]` section (D-05).
    #[must_use]
    pub fn is_empty(&self) -> bool {
        self.tables.is_empty() && self.buckets.is_empty() && self.statements.is_empty()
    }
}

/// DynamoDB table permission (sugar form).
///
/// `actions` is a subset of `{"read", "write", "readwrite"}`. Translation
/// happens in `cargo_pmcp::deployment::iam::render_table` (added in Wave 3)
/// and emits the 4-action DynamoDB lists per CONTEXT.md D-02:
/// - `read` → `GetItem`, `Query`, `Scan`, `BatchGetItem`
/// - `write` → `PutItem`, `UpdateItem`, `DeleteItem`, `BatchWriteItem`
/// - `readwrite` → union (8 actions)
///
/// When `include_indexes = true`, the emitted resource list also contains
/// `arn:aws:dynamodb:${this.region}:${this.account}:table/NAME/index/*`.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct TablePermission {
    /// Table name (AWS region + account inherited from the deploy context).
    pub name: String,

    /// Sugar keywords: subset of `{"read", "write", "readwrite"}`.
    /// Unknown values are rejected by the Wave 4 validator.
    pub actions: Vec<String>,

    /// When true, grants to `table/NAME/index/*` (GSI/LSI access).
    #[serde(default)]
    pub include_indexes: bool,
}

/// S3 bucket permission (sugar form — object-level ARN only).
///
/// Translation happens in `cargo_pmcp::deployment::iam::render_bucket`
/// (added in Wave 3):
/// - `read` → `GetObject`
/// - `write` → `PutObject`, `DeleteObject`
/// - `readwrite` → union (3 actions)
///
/// Resource ARN is `arn:aws:s3:::NAME/*` (object-level). Bucket-level
/// operations (e.g. `s3:ListBucket`) must be declared via
/// `[[iam.statements]]` per CR scope.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct BucketPermission {
    /// Bucket name.
    pub name: String,

    /// Sugar keywords: subset of `{"read", "write", "readwrite"}`.
    pub actions: Vec<String>,
}

/// Raw IAM PolicyStatement (passthrough after validation).
///
/// Validated in Wave 4 (validator module) before translation. Emitted
/// verbatim as `new iam.PolicyStatement({effect, actions, resources})` in
/// the generated stack.ts.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct IamStatement {
    /// `"Allow"` or `"Deny"`. Validated in Wave 4.
    pub effect: String,

    /// Non-empty list of actions matching `^[a-z0-9-]+:[A-Za-z0-9*]+$`.
    /// Validator enforces regex and rejects Allow `*:*` + resources `*`.
    pub actions: Vec<String>,

    /// Non-empty list of ARN or `*` resources. Cross-account ARNs emit a warning.
    pub resources: Vec<String>,
}
```

```rust
//! Phase 76 Wave 2: IamConfig serde integration tests.
//!
//! Exercises the public API from an integration-test crate — ensures that
//! downstream users (including Wave 3 translation and Wave 4 validation) can
//! rely on IamConfig being deserialize-able via the standard `toml` crate and
//! accessible through `cargo_pmcp::deployment::config::...` re-exports.

use cargo_pmcp::deployment::config::{
    BucketPermission, DeployConfig, IamConfig, IamStatement, TablePermission,
};

const MINIMAL_DEPLOY_TOML: &str = r#"
[target]
type = "pmcp-run"

[aws]
region = "us-west-2"

[server]
name = "demo-server"
"#;

const COST_COACH_DEPLOY_TOML: &str = r#"
[target]
type = "pmcp-run"

[aws]
region = "us-west-2"

[server]
name = "cost-coach"

[[iam.tables]]
name = "cost-coach-tenants"
actions = ["readwrite"]
include_indexes = true

[[iam.buckets]]
name = "cost-coach-snapshots"
actions = ["readwrite"]

[[iam.statements]]
effect = "Allow"
actions = ["secretsmanager:GetSecretValue"]
resources = ["arn:aws:secretsmanager:us-west-2:*:secret:cost-coach/*"]
"#;

#[test]
fn minimal_deploy_toml_has_empty_iam() {
    let cfg: DeployConfig = toml::from_str(MINIMAL_DEPLOY_TOML).expect("parses");
    assert!(cfg.iam.is_empty(), "minimal config must have empty IamConfig");
    assert!(cfg.iam.tables.is_empty());
    assert!(cfg.iam.buckets.is_empty());
    assert!(cfg.iam.statements.is_empty());
}

#[test]
fn cost_coach_toml_parses_expected_shape() {
    let cfg: DeployConfig = toml::from_str(COST_COACH_DEPLOY_TOML).expect("parses");

    assert_eq!(cfg.iam.tables.len(), 1);
    let table = &cfg.iam.tables[0];
    assert_eq!(table.name, "cost-coach-tenants");
    assert_eq!(table.actions, vec!["readwrite".to_string()]);
    assert!(table.include_indexes);

    assert_eq!(cfg.iam.buckets.len(), 1);
    let bucket = &cfg.iam.buckets[0];
    assert_eq!(bucket.name, "cost-coach-snapshots");
    assert_eq!(bucket.actions, vec!["readwrite".to_string()]);

    assert_eq!(cfg.iam.statements.len(), 1);
    let stmt = &cfg.iam.statements[0];
    assert_eq!(stmt.effect, "Allow");
    assert_eq!(stmt.actions, vec!["secretsmanager:GetSecretValue".to_string()]);
    assert_eq!(
        stmt.resources,
        vec!["arn:aws:secretsmanager:us-west-2:*:secret:cost-coach/*".to_string()]
    );

    assert!(!cfg.iam.is_empty());
}

#[test]
fn empty_iam_does_not_serialize_iam_section() {
    // D-05 backward compat invariant: a DeployConfig with no iam section must
    // serialize WITHOUT the [iam] / [[iam.tables]] / etc tables.
    let cfg: DeployConfig = toml::from_str(MINIMAL_DEPLOY_TOML).expect("parses");
    let out = toml::to_string(&cfg).expect("serializes");
    assert!(
        !out.contains("[iam]") && !out.contains("[[iam.tables]]")
            && !out.contains("[[iam.buckets]]") && !out.contains("[[iam.statements]]"),
        "empty IamConfig must not emit any iam table (D-05 violation)\nOutput was:\n{out}"
    );
}

#[test]
fn populated_iam_roundtrips_losslessly() {
    // Round trip: parse → serialize → parse → compare structurally.
    // No PartialEq derive on IamConfig (PATTERNS.md §S1), so compare field-by-field.
    let orig: DeployConfig = toml::from_str(COST_COACH_DEPLOY_TOML).expect("parses");
    let serialized = toml::to_string(&orig).expect("serializes");
    let reparsed: DeployConfig = toml::from_str(&serialized)
        .unwrap_or_else(|e| panic!("reparse failed — serialized:\n{serialized}\nerror: {e}"));

    assert_eq!(orig.iam.tables.len(), reparsed.iam.tables.len());
    assert_eq!(orig.iam.tables[0].name, reparsed.iam.tables[0].name);
    assert_eq!(orig.iam.tables[0].actions, reparsed.iam.tables[0].actions);
    assert_eq!(orig.iam.tables[0].include_indexes, reparsed.iam.tables[0].include_indexes);

    assert_eq!(orig.iam.buckets.len(), reparsed.iam.buckets.len());
    assert_eq!(orig.iam.buckets[0].name, reparsed.iam.buckets[0].name);
    assert_eq!(orig.iam.buckets[0].actions, reparsed.iam.buckets[0].actions);

    assert_eq!(orig.iam.statements.len(), reparsed.iam.statements.len());
    assert_eq!(orig.iam.statements[0].effect, reparsed.iam.statements[0].effect);
    assert_eq!(orig.iam.statements[0].actions, reparsed.iam.statements[0].actions);
    assert_eq!(orig.iam.statements[0].resources, reparsed.iam.statements[0].resources);
}

}
```
