//! IAM translation rules: IamConfig ã TypeScript addToRolePolicy strings.
//!
//! Produces 4-space-indented `mcpFunction.addToRolePolicy(new iam.PolicyStatement(...))`
//! calls that match the existing template's emission style (`init.rs:594, 607`).
//! Wave 4 will add validation in a separate `validate` function that MUST run before
//! `render_iam_block` in the CLI entry points.
//!
//! Translation rules are locked per CONTEXT.md D-02:
//!  - `read`  ã `GetItem`, `Query`, `Scan`, `BatchGetItem`      (4 actions)
//!  - `write` ã `PutItem`, `UpdateItem`, `DeleteItem`, `BatchWriteItem` (4 actions)
//!  - `readwrite` ã union (8 actions)
//!
//! Resource ARNs use `${this.region}` / `${this.account}` CDK placeholders so the
//! emitted template inherits the deploy-target's region/account at synth time.

use std::fmt::Write;

use crate::deployment::config::{BucketPermission, IamConfig, IamStatement, TablePermission};

/// Render the full IAM block. Returns `""` when `iam.is_empty()` ã this preserves
/// D-05 backward-compat byte-identity for configs without an `[iam]` section.
///
/// Output is zero or more `addToRolePolicy(...)` statements separated by `\n\n`,
/// each 4-space-indented. Ordering: all tables, then buckets, then raw statements
/// (RESEARCH.md Q4 ã locks golden-file determinism).
#[must_use]
pub fn render_iam_block(iam: &IamConfig) -> String {
    if iam.is_empty() {
        return String::new();
    }

    let mut out = String::new();
    // Leading blank line + comment separator to set the operator block off from
    // the platform's built-in composition policies (init.rs:594, 607).
    out.push_str("\n    // ========================================================================\n");
    out.push_str("    // Operator-declared IAM (from .pmcp/deploy.toml [iam])\n");
    out.push_str("    // ========================================================================\n");

    for table in &iam.tables {
        out.push_str(&render_table(table));
        out.push('\n');
    }
    for bucket in &iam.buckets {
        out.push_str(&render_bucket(bucket));
        out.push('\n');
    }
    for stmt in &iam.statements {
        out.push_str(&render_statement(stmt));
        out.push('\n');
    }

    out
}

fn render_table(t: &TablePermission) -> String {
    let has_read = t.actions.iter().any(|a| a == "read" || a == "readwrite");
    let has_write = t.actions.iter().any(|a| a == "write" || a == "readwrite");

    let mut actions: Vec<&str> = Vec::with_capacity(8);
    if has_read {
        actions.extend_from_slice(&[
            "dynamodb:GetItem",
            "dynamodb:Query",
            "dynamodb:Scan",
            "dynamodb:BatchGetItem",
        ]);
    }
    if has_write {
        actions.extend_from_slice(&[
            "dynamodb:PutItem",
            "dynamodb:UpdateItem",
            "dynamodb:DeleteItem",
            "dynamodb:BatchWriteItem",
        ]);
    }

    let actions_ts = format_action_array(&actions);
    let mut resources = format!(
        "`arn:aws:dynamodb:${{this.region}}:${{this.account}}:table/{}`",
        t.name
    );
    if t.include_indexes {
        write!(
            resources,
            ",\n        `arn:aws:dynamodb:${{this.region}}:${{this.account}}:table/{}/index/*`",
            t.name
        )
        .expect("writing to String is infallible");
    }

    format!(
        "    mcpFunction.addToRolePolicy(new iam.PolicyStatement({{\n\
         \x20     effect: iam.Effect.ALLOW,\n\
         \x20     actions: {actions_ts},\n\
         \x20     resources: [\n        {resources},\n      ],\n\
         \x20   }}));\n"
    )
}

fn render_bucket(b: &BucketPermission) -> String {
    let has_read = b.actions.iter().any(|a| a == "read" || a == "readwrite");
    let has_write = b.actions.iter().any(|a| a == "write" || a == "readwrite");

    let mut actions: Vec<&str> = Vec::with_capacity(3);
    if has_read {
        actions.push("s3:GetObject");
    }
    if has_write {
        actions.push("s3:PutObject");
        actions.push("s3:DeleteObject");
    }

    let actions_ts = format_action_array(&actions);
    let resources = format!("`arn:aws:s3:::{}/*`", b.name);

    format!(
        "    mcpFunction.addToRolePolicy(new iam.PolicyStatement({{\n\
         \x20     effect: iam.Effect.ALLOW,\n\
         \x20     actions: {actions_ts},\n\
         \x20     resources: [\n        {resources},\n      ],\n\
         \x20   }}));\n"
    )
}

fn render_statement(s: &IamStatement) -> String {
    let effect = if s.effect.eq_ignore_ascii_case("Deny") {
        "iam.Effect.DENY"
    } else {
        "iam.Effect.ALLOW" // default ã validator catches non-Allow/Deny in Wave 4
    };

    let actions_ts = format_action_array_owned(&s.actions);
    let resources_ts = format_resource_array(&s.resources);

    format!(
        "    mcpFunction.addToRolePolicy(new iam.PolicyStatement({{\n\
         \x20     effect: {effect},\n\
         \x20     actions: {actions_ts},\n\
         \x20     resources: {resources_ts},\n\
         \x20   }}));\n"
    )
}

fn format_action_array(actions: &[&str]) -> String {
    if actions.is_empty() {
        return "[]".to_string();
    }
    let items = actions
        .iter()
        .map(|a| format!("'{a}'"))
        .collect::<Vec<_>>()
        .join(", ");
    format!("[{items}]")
}

fn format_action_array_owned(actions: &[String]) -> String {
    if actions.is_empty() {
        return "[]".to_string();
    }
    let items = actions
        .iter()
        .map(|a| format!("'{a}'"))
        .collect::<Vec<_>>()
        .join(", ");
    format!("[{items}]")
}

fn format_resource_array(resources: &[String]) -> String {
    if resources.is_empty() {
        return "[]".to_string();
    }
    let items = resources
        .iter()
        .map(|r| {
            // Resources may be `*` or an ARN; quote as TS string literal.
            format!("'{r}'")
        })
        .collect::<Vec<_>>()
        .join(", ");
    format!("[{items}]")
}

//! Strategies draw sugar keywords from `{"read", "write", "readwrite"}` and valid

fn arb_sugar_action() -> impl Strategy<Value = String> {
    prop_oneof![
        Just("read".to_string()),
        Just("write".to_string()),
        Just("readwrite".to_string()),
    ]

fn arb_sugar_actions_vec() -> impl Strategy<Value = Vec<String>> {
