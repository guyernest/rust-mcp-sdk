use anyhow::{anyhow, Result};
use regex::Regex;

/// A non-blocking validation finding. Hard errors short-circuit via `Result::Err`.
#[derive(Debug, Clone)]
pub struct Warning {
    pub message: String,
}

impl Warning {
    fn new(message: impl Into<String>) -> Self {
        Self { message: message.into() }
    }
}

const ACTION_REGEX: &str = r"^[a-z0-9-]+:[A-Za-z0-9*]+$";
const VALID_SUGAR: &[&str] = &["read", "write", "readwrite"];

/// Curated list of well-known AWS service prefixes. Unknown prefixes trigger a
/// warning (not an error) per CONTEXT.md §Validation. Updated annually or on
/// user report.

/// Validate IAM declarations per CR-locked rules. See module docs for rule list.
///
/// Hard errors → `Err`. Warnings → `Ok(warnings)` — caller is expected to surface
/// them to stderr.
///
/// # Errors
/// Returns `Err` on any of the 6 hard-error rules in CONTEXT.md §Validation.
pub fn validate(iam: &IamConfig) -> Result<Vec<Warning>> {
    let action_re = Regex::new(ACTION_REGEX).expect("ACTION_REGEX is a static, known-good pattern");
    let mut warnings = Vec::new();

    // Rule 6: empty table names.
    for (idx, t) in iam.tables.iter().enumerate() {
        if t.name.trim().is_empty() {
            return Err(anyhow!(
                "[iam.tables][{idx}]: name must not be empty"
            ));
        }
        // Rule 5: sugar keywords.
        for a in &t.actions {
            if !VALID_SUGAR.contains(&a.as_str()) {
                return Err(anyhow!(
                    "[iam.tables][{idx}] '{}': unknown sugar keyword '{}' — allowed: {{read, write, readwrite}}",
                    t.name, a
                ));
            }
        }
        if t.actions.is_empty() {
            return Err(anyhow!(
                "[iam.tables][{idx}] '{}': actions must not be empty",
                t.name
            ));
        }
    }

    // Rule 6 (buckets): empty bucket names.
    for (idx, b) in iam.buckets.iter().enumerate() {
        if b.name.trim().is_empty() {
            return Err(anyhow!(
                "[iam.buckets][{idx}]: name must not be empty"
            ));
        }
        for a in &b.actions {
            if !VALID_SUGAR.contains(&a.as_str()) {
                return Err(anyhow!(
                    "[iam.buckets][{idx}] '{}': unknown sugar keyword '{}' — allowed: {{read, write, readwrite}}",
                    b.name, a
                ));
            }
        }
        if b.actions.is_empty() {
            return Err(anyhow!(
                "[iam.buckets][{idx}] '{}': actions must not be empty",
                b.name
            ));
        }
    }

    // [[iam.statements]] rules.
    for (idx, stmt) in iam.statements.iter().enumerate() {
        // Rule 2: effect must be Allow or Deny.
        if stmt.effect != "Allow" && stmt.effect != "Deny" {
            return Err(anyhow!(
                "[iam.statements][{idx}]: effect must be 'Allow' or 'Deny', got '{}'",
                stmt.effect
            ));
        }

        // Rule 3: empty actions.
        if stmt.actions.is_empty() {
            return Err(anyhow!(
                "[iam.statements][{idx}]: actions must not be empty"
            ));
        }

        // Rule 3: empty resources.
        if stmt.resources.is_empty() {
            return Err(anyhow!(
                "[iam.statements][{idx}]: resources must not be empty"
            ));
        }

        // Rule 1: Allow + actions=["*"] + resources=["*"] hard error.
        if stmt.effect == "Allow"
            && stmt.actions.len() == 1
            && stmt.actions[0] == "*"
            && stmt.resources.len() == 1
            && stmt.resources[0] == "*"
        {
            return Err(anyhow!(
                "[iam.statements][{idx}]: Allow + actions=[\"*\"] + resources=[\"*\"] is a wildcard escalation footgun — refuse to deploy. Tighten actions and resources, or use [[iam.tables]] / [[iam.buckets]] sugar blocks."
            ));
        }

        // Rule 4: action regex.
        for a in &stmt.actions {
            if !action_re.is_match(a) {
                return Err(anyhow!(
                    "[iam.statements][{idx}]: action '{a}' does not match {ACTION_REGEX}"
                ));
            }
            // Rule 7: warning on unknown service prefix.
            if let Some((prefix, _)) = a.split_once(':') {
                if !KNOWN_SERVICE_PREFIXES.contains(&prefix) {
                    warnings.push(Warning::new(format!(
                        "[iam.statements][{idx}]: unknown service prefix '{prefix}' in action '{a}' — verify this is a valid AWS service"
                    )));
                }
            }
        }

        // Rule 8: cross-account ARN warning (best-effort: we don't know the
        // deploy-target account from IamConfig alone, so warn only when the
        // account field is a literal 12-digit AWS account that's clearly not
        // the `*` wildcard). A full cross-account check requires passing
        // aws.account through to validate — deferred to an optional overload.
        for r in &stmt.resources {
            if let Some(acct) = extract_account_from_arn(r) {
                if acct.chars().all(|c| c.is_ascii_digit()) && acct.len() == 12 {
                    warnings.push(Warning::new(format!(
                        "[iam.statements][{idx}]: resource '{r}' pins a specific AWS account '{acct}' — verify this matches your deploy target (use '*' or omit the account segment for account-agnostic ARNs)"
                    )));
                }
            }
        }
    }

    Ok(warnings)
}

/// Extract the `account` segment (index 4) of an ARN `arn:partition:service:region:account:resource`.
/// Returns None if the input is not shaped like an ARN.
fn extract_account_from_arn(arn: &str) -> Option<&str> {
    let mut parts = arn.splitn(6, ':');
    let _arn = parts.next()?;      // "arn"
    let _partition = parts.next()?; // "aws" / "aws-cn" / "aws-us-gov"
    if _arn != "arn" {
        return None;
    }
    let _service = parts.next()?;
    let _region = parts.next()?;
    let account = parts.next()?;
    let _resource = parts.next()?;
    Some(account)
}

/// Validate the deployment configuration (.pmcp/deploy.toml) — IAM focus.
///
/// This is the pre-flight equivalent of the same validation that runs inside
/// `cargo pmcp deploy`. Returns `Ok(())` on success (even with warnings);
/// returns `Err` on any hard-error rule per CR.
pub fn validate_deploy(server: Option<String>, verbose: bool) -> Result<()> {
    let not_quiet = std::env::var("PMCP_QUIET").is_err();

    let project_root = match server {
        Some(path) => std::path::PathBuf::from(path),
        None => std::env::current_dir().context("failed to read current directory")?,
    };

    if not_quiet {
        println!("\n{}", style("PMCP Deploy Config Validation").cyan().bold());
        println!("{}", style("━".repeat(50)).dim());
        if verbose {
            println!("  Project: {}", project_root.display());
        }
    }

    let config = crate::deployment::config::DeployConfig::load(&project_root)
        .context("failed to load .pmcp/deploy.toml")?;

    let warnings = crate::deployment::iam::validate(&config.iam)
        .context("IAM validation failed")?;

    if not_quiet {
        for w in &warnings {
            eprintln!("  {} {}", style("warning:").yellow(), w.message);
        }
        if warnings.is_empty() {
            println!("  {} IAM configuration valid (no warnings)", style("✓").green());
        } else {
            println!(
                "  {} IAM configuration valid ({} warning{})",
                style("✓").green(),
                warnings.len(),
                if warnings.len() == 1 { "" } else { "s" }
            );
        }
    }

    Ok(())
}
