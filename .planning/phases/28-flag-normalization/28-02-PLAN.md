```rust
use clap::{Args, ValueEnum};
use std::path::PathBuf;

#[derive(Debug, Clone, ValueEnum)]
pub enum FormatValue {
    Text,
    Json,
}

#[derive(Debug, Args)]
pub struct OutputFlags {
    #[arg(long, short)]
    pub output: Option<PathBuf>,
}

#[derive(Debug, Args)]
pub struct FormatFlags {
    #[arg(long, value_enum, default_value = "text")]
    pub format: FormatValue,
}

/// Shared flags for commands that connect to an MCP server via URL or --server.
/// Use via #[command(flatten)] on commands where URL is optional (has --server fallback).
#[derive(Debug, Args)]
pub struct ServerFlags {
    /// MCP server URL (positional)
    #[arg(index = 1)]
    pub url: Option<String>,

    /// Server name on pmcp.run (alternative to URL)
    #[arg(long)]
    pub server: Option<String>,
}
```
