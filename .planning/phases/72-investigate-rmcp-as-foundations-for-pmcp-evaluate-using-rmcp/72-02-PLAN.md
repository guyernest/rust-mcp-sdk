```rust
//! Phase 72 throwaway spike. This file lives ONLY on branch spike/72-poc-slice-1.
//! The branch and this directory are deleted at the end of Plan 02 Task 1b.
//!
//! Goal: measure (a) compile errors (b) LOC delta when referencing rmcp::model types
//! directly instead of pmcp-owned equivalents.

use rmcp::model::{
    JsonRpcRequest as RmcpRequest,
    JsonRpcResponse as RmcpResponse,
    RequestId as RmcpRequestId,
};

fn main() {
    // Construct a minimal JSON-RPC request using rmcp types.
    let req_json = serde_json::json!({
        "jsonrpc": "2.0",
        "id": 1,
        "method": "ping",
        "params": null
    });

    // Roundtrip test: does rmcp's JsonRpcRequest parse the canonical shape?
    let req: RmcpRequest = serde_json::from_value(req_json.clone())
        .expect("rmcp::model::JsonRpcRequest should parse canonical shape");

    // Roundtrip serialize.
    let roundtripped = serde_json::to_value(&req)
        .expect("rmcp::model::JsonRpcRequest should serialize");

    assert_eq!(req_json, roundtripped, "serde roundtrip diverged");
    println!("PASS: serde roundtrip holds for rmcp::model::JsonRpcRequest");
}
```
