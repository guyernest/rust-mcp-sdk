        let validator = AppValidator::new(mode, None);
        validator.validate_widgets(&[(

            "ui://test".to_string(),
            html.to_string(),
        )])

    fn count_status(results: &[mcp_tester::TestResult], wanted: TestStatus) -> usize {
        results.iter().filter(|r| r.status == wanted).count()

        results
