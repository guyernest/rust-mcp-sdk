Add `extract_inline_scripts`. This is cog ≤8: one regex iteration loop, one type-attribute filter (Pitfall 3). The `attrs` named capture lets us reject `type="application/json"` and `src=`-bearing tags:

```rust
/// Concatenate the bodies of all inline `<script>` tags except those with
/// `type="application/json"` or `src=` attribute. Returns a single String for
/// downstream regex sweeps.
fn extract_inline_scripts(html: &str) -> String {
    let mut out = String::new();
    for cap in script_block_re().captures_iter(html) {
        let attrs = cap.name("attrs").map_or("", |m| m.as_str());
        if attrs.contains("application/json") || attrs.contains("src=") {
            continue;
        }
        if let Some(body) = cap.name("body") {
            out.push_str(body.as_str());
            out.push('\n');
        }
    }
    out
}
```

```rust
/// Run the regex sweep over a widget body and return signal flags.
fn scan_widget(html: &str) -> WidgetSignals {
    let scripts = extract_inline_scripts(html);
    let mut handlers_present: Vec<&'static str> = Vec::new();
    if handler_onteardown_re().is_match(&scripts) {
        handlers_present.push("onteardown");
    }
    if handler_ontoolinput_re().is_match(&scripts) {
        handlers_present.push("ontoolinput");
    }
    if handler_ontoolcancelled_re().is_match(&scripts) {
        handlers_present.push("ontoolcancelled");
    }
    if handler_onerror_re().is_match(&scripts) {
        handlers_present.push("onerror");
    }
    WidgetSignals {
        has_ext_apps_import: ext_apps_import_re().is_match(&scripts),
        has_new_app: new_app_call_re().is_match(&scripts),
        has_connect: connect_call_re().is_match(&scripts),
        has_chatgpt_only_channels: chatgpt_only_channels_re().is_match(&scripts),
        handlers_present,
        has_ontoolresult: handler_ontoolresult_re().is_match(&scripts),
    }
}
```

```rust
impl AppValidator {
    /// Validate inline widget HTML for Claude Desktop / MCP Apps SDK wiring.
    ///
    /// Pure function: takes already-fetched widget bodies and returns
    /// `TestResult`s. Severity is mode-aware: `Standard` and `ChatGpt` emit
    /// `Warning`, `ClaudeDesktop` emits `Failed`.
    ///
    /// Pass `(uri, html)` tuples; one TestResult per (widget, signal-kind).
    pub fn validate_widgets(
        &self,
        widget_bodies: &[(String, String)],
    ) -> Vec<TestResult> {
        let mut results = Vec::new();
        for (uri, html) in widget_bodies {
            let signals = scan_widget(html);
            results.extend(self.emit_results_for(uri, &signals));
        }
        results
    }

    fn emit_results_for(&self, uri: &str, s: &WidgetSignals) -> Vec<TestResult> {
        let mut out = Vec::new();
        // SDK presence: import literal OR >=3 of 4 handler assignments
        let sdk_present = s.has_ext_apps_import || s.handlers_present.len() >= 3;
        out.push(self.widget_result(
            uri,
            "MCP Apps SDK wiring",
            sdk_present,
            "Widget does not import @modelcontextprotocol/ext-apps and does not register >=3 of the 4 protocol handlers. [guide:handlers-before-connect]",
        ));
        // new App({...})
        out.push(self.widget_result(
            uri,
            "App constructor",
            s.has_new_app,
            "Widget does not call `new App({...})`. [guide:handlers-before-connect]",
        ));
        // Required handlers (each is its own row so error messages name them)
        for name in ["onteardown", "ontoolinput", "ontoolcancelled", "onerror"] {
            let present = s.handlers_present.iter().any(|h| *h == name);
            out.push(self.widget_result(
                uri,
                &format!("handler: {name}"),
                present,
                &format!("Widget does not register `app.{name}` before `connect()`. [guide:handlers-before-connect]"),
            ));
        }
        // ontoolresult is soft (Warning even in ClaudeDesktop)
        out.push(self.widget_ontoolresult_result(uri, s));
        // connect()
        out.push(self.widget_result(
            uri,
            "connect() call",
            s.has_connect,
            "Widget does not call `app.connect()`. [guide:handlers-before-connect]",
        ));
        // ChatGPT-only channels: ERROR in ClaudeDesktop, OK in ChatGpt/Standard
        if s.has_chatgpt_only_channels && !s.has_ext_apps_import && s.handlers_present.is_empty() {
            out.push(self.widget_chatgpt_only_result(uri));
        }
        out
    }

    fn widget_result(&self, uri: &str, label: &str, present: bool, missing_details: &str) -> TestResult {
        TestResult {
            name: format!("[{uri}] {label}"),
            category: TestCategory::Apps,
            status: widget_status_for_mode(self.mode, present),
            duration: Duration::from_secs(0),
            error: None,
            details: if present { None } else { Some(missing_details.to_string()) },
        }
    }

    fn widget_ontoolresult_result(&self, uri: &str, s: &WidgetSignals) -> TestResult {
        // ontoolresult is always soft (Warning), regardless of mode, per
        // RESEARCH Locked Decision 3 (some widgets render from
        // getHostContext().toolOutput).
        TestResult {
            name: format!("[{uri}] handler: ontoolresult"),
            category: TestCategory::Apps,
            status: if s.has_ontoolresult {
                TestStatus::Passed
            } else {
                TestStatus::Warning
            },
            duration: Duration::from_secs(0),
            error: None,
            details: if s.has_ontoolresult {
                None
            } else {
                Some("Widget does not register `app.ontoolresult` (soft warning — may render from getHostContext().toolOutput). [guide:handlers-before-connect]".to_string())
            },
        }
    }

    fn widget_chatgpt_only_result(&self, uri: &str) -> TestResult {
        // ChatGPT-only channels with no ext-apps wiring: ERROR in ClaudeDesktop,
        // OK in chatgpt/standard.
        let status = match self.mode {
            AppValidationMode::ClaudeDesktop => TestStatus::Failed,
            AppValidationMode::ChatGpt | AppValidationMode::Standard => TestStatus::Passed,
        };
        TestResult {
            name: format!("[{uri}] chatgpt-only channels detected"),
            category: TestCategory::Apps,
            status,
            duration: Duration::from_secs(0),
```

| widget HTML body → scanner | Untrusted input: widget body is fetched over the network from arbitrary MCP servers (Plan 02 wires this). The validator function is `pure` (no IO), so no privileged state crosses this boundary, but availability of the CLI `cargo pmcp test apps` is at risk. |

|-----------|----------|-----------|-------------|-----------------|
| T-78-01 | Denial of Service | `scan_widget` regex sweep on attacker-supplied HTML | mitigate | The `regex` crate uses a linear-time NFA — no catastrophic backtracking by design (CITED docs.rs/regex#performance). All 10 patterns are anchored to short literals (`\.\s*onteardown\s*=`, etc.) bounded by character classes, no unbounded `.*` between two `.*` patterns. Property test in Plan 03 (`prop_scan_never_panics`) and fuzz target (`app_widget_scanner`) provide ongoing mitigation. |
| T-78-02 | Denial of Service | `extract_inline_scripts` on enormous widget body (memory exhaustion) | mitigate | Plan 02's `read_widget_bodies` enforces a soft 10MB cap before passing the body to the validator (per RESEARCH §Security Domain table line 570). Within Plan 01: `extract_inline_scripts` allocates a single `String` proportional to input size — no quadratic growth. |
| T-78-03 | Denial of Service | `scan_widget` panics on adversarial input | mitigate | `OnceLock<Regex>::get_or_init` with `.unwrap()` is safe because all regexes are static literals validated at first call (Task 1 includes `regexes_compile` test).
