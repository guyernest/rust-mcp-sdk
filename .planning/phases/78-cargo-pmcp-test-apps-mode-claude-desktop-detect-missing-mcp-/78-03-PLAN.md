```rust
//! Property-based tests for `AppValidator::validate_widgets`.
//!
//! Uses proptest to verify invariants across randomized inputs:
//! - Scanner does not panic on arbitrary printable input (PHASE-78-VALIDATION
//!   row "Scanner robustness")
//! - Whitespace normalization is idempotent (the count and distribution of
//!   TestStatus values does not depend on extra whitespace)

use mcp_tester::{AppValidationMode, AppValidator, TestStatus};
use proptest::prelude::*;

fn run_validator(html: &str) -> Vec<TestStatus> {
    let validator = AppValidator::new(AppValidationMode::ClaudeDesktop, None);
    validator
        .validate_widgets(&[("ui://prop-test".to_string(), html.to_string())])
        .into_iter()
        .map(|r| r.status)
        .collect()
}

proptest! {
    /// Scanner must not panic on arbitrary printable Unicode input bounded to 4096 chars.
    /// `\PC` = any character that is NOT a control character.
    #[test]
    fn prop_scan_never_panics(html in r"\PC{0,4096}") {
        let _ = run_validator(&html);
    }

    /// Whitespace doubling does not change the result-status distribution.
    /// Bounded alphabet ensures we exercise the regex without tripping on
    /// pathological character classes. Per Info 9, the alphabet excludes the
    /// embedded double-quote (the proptest regex strategy treats `\x22`
    /// literally in raw strings, which doesn't compile cleanly — drop the
    /// quote entirely; whitespace + tag-like fragments still get exercised).
    #[test]
    fn prop_whitespace_idempotent(html in r"[a-zA-Z<>/= .]{0,500}") {
        let s1 = run_validator(&html);
        let html2 = html.replace(' ', "  ").replace('\n', "\n\n");
        let s2 = run_validator(&html2);
        prop_assert_eq!(s1.len(), s2.len(), "result vec length must be stable under whitespace doubling");
        let count1_failed = s1.iter().filter(|s| **s == TestStatus::Failed).count();
```

```rust
//! Example: Validate a broken vs. corrected widget pair under --mode claude-desktop.
//!
//! This is the ALWAYS-requirement working example for Phase 78. It demonstrates

//!
//! Usage:

use mcp_tester::{AppValidationMode, AppValidator, OutputFormat, TestReport, TestStatus};

const BROKEN: &str = include_str!("../tests/fixtures/widgets/broken_no_sdk.html");
const CORRECTED: &str = include_str!("../tests/fixtures/widgets/corrected_minimal.html");

fn run_one(label: &str, html: &str) {
    println!("\n=== {label} (mode = claude-desktop) ===\n");
    let validator = AppValidator::new(AppValidationMode::ClaudeDesktop, None);
    let results =
        validator.validate_widgets(&[(format!("ui://example-{label}"), html.to_string())]);
    let failed = results
        .iter()
        .filter(|r| r.status == TestStatus::Failed)
        .count();
    let warnings = results
        .iter()
        .filter(|r| r.status == TestStatus::Warning)
        .count();
    let passed = results
        .iter()
        .filter(|r| r.status == TestStatus::Passed)
        .count();

    let mut report = TestReport::new();
    for r in results {
        report.add_test(r);
    }
    report.print(OutputFormat::Pretty);
    println!(
        "\nSummary for {label}: {passed} passed, {warnings} warnings, {failed} failed.\n"
    );
}

fn main() {
    println!("MCP Apps widget validator — broken/corrected demo");
    println!("==================================================");
    run_one("broken_no_sdk", BROKEN);
    run_one("corrected_minimal", CORRECTED);
    println!("Done. The broken widget produced Failed rows; the corrected one did not.");
}
```
