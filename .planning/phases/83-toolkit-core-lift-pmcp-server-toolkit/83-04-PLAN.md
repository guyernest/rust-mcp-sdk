**Step D — implement `validate()` and `from_toml_strict_validated()` on `ServerConfig`:**

```rust
impl ServerConfig {
    /// Parse `ServerConfig` from a TOML config string.
    ///
    /// Returns `ToolkitError::Parse` on syntax error or unknown field (per D-13
    /// `deny_unknown_fields` discipline). Does NOT enforce semantic validation —
    /// callers wanting required-field guarantees should use
    /// [`Self::from_toml_strict_validated`] instead.
    ///
    /// # Example
    ///
    /// ```
    /// use pmcp_server_toolkit::config::ServerConfig;
    ///
    /// let toml = r#"
    ///     [server]
    ///     id = "demo"
    ///     name = "Demo"
    ///     version = "0.1.0"
    /// "#;
    /// let cfg = ServerConfig::from_toml(toml).expect("parse");
    /// assert_eq!(cfg.server.name, "Demo");
    /// ```
    pub fn from_toml(toml_str: &str) -> crate::error::Result<Self> {
        toml::from_str(toml_str).map_err(crate::error::ToolkitError::Parse)
    }

    /// Parse + validate. Per review R8 — guards against missing-required-value
    /// bugs that the `Default` impls on sub-sections would otherwise hide
    /// behind silent empty strings.
    ///
    /// # Errors
    ///
    /// Returns `ToolkitError::Parse` on TOML syntax / unknown-field errors, or
    /// `ToolkitError::Validation` on missing required values (empty server.name,
    /// empty server.version, empty tool name, empty table name).
    ///
    /// # Example
    ///
    /// ```
    /// use pmcp_server_toolkit::config::ServerConfig;
    /// let toml = r#"
    ///     [server]
    ///     name = "demo"
    ///     version = "0.1.0"
    /// "#;
    /// let cfg = ServerConfig::from_toml_strict_validated(toml).expect("valid");
    /// # let _ = cfg;
    /// ```
    pub fn from_toml_strict_validated(toml_str: &str) -> crate::error::Result<Self> {
        let cfg = Self::from_toml(toml_str)?;
        cfg.validate()?;
        Ok(cfg)
    }

    /// Validate required-field semantics that `#[serde(default)]` would
    /// otherwise mask. Per review R8.
    ///
    /// # Errors
    ///
    /// Returns `ConfigValidationError::*` for each rule violated.
    pub fn validate(&self) -> Result<(), crate::error::ConfigValidationError> {
        use crate::error::ConfigValidationError as E;
        if self.server.name.trim().is_empty() {
            return Err(E::EmptyServerName);
        }
        if self.server.version.trim().is_empty() {
            return Err(E::EmptyServerVersion);
        }
        for (i, tool) in self.tools.iter().enumerate() {
            if tool.name.trim().is_empty() {
                return Err(E::EmptyToolName(i));
            }
        }
        for (i, table) in self.database.tables.iter().enumerate() {
            if table.name.trim().is_empty() {
                return Err(E::EmptyTableName(i));
            }
        }
        Ok(())
    }
}
```
