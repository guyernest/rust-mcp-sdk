//! [[tools]] → ToolInfo synthesizer (TKIT-07).
//!
//! Net-new code for Phase 83. Turns curated config entries into complete pmcp
//! ToolInfo + Arc<dyn ToolHandler> pairs with zero per-tool Rust handlers.

use std::sync::Arc;

use async_trait::async_trait;
use pmcp::server::ToolHandler;
use pmcp::types::{Content, ToolAnnotations, ToolInfo};
use pmcp::{CallToolResult, RequestHandlerExtra};
use serde_json::{Map, Value, json};

use crate::config::{AnnotationsDecl, ParamDecl, ServerConfig, ToolDecl};
use crate::error::{Result, ToolkitError};

/// Synthesize one `ToolInfo` + handler per `[[tools]]` config entry.
///
/// Per CONTEXT.md D-10, returns owned data the caller registers via
/// `pmcp::ServerBuilder::tool_arc(name, handler)`.
///
/// # Errors
///
/// Returns `ToolkitError::Synth(...)` if a tool declaration is internally
/// inconsistent (e.g. duplicate names, malformed default values).
///
/// # Example
///
/// ```
/// use pmcp_server_toolkit::config::ServerConfig;
/// use pmcp_server_toolkit::tools::synthesize_from_config;
///
/// let cfg = ServerConfig::default();
/// let synthesized = synthesize_from_config(&cfg).unwrap();
/// assert_eq!(synthesized.len(), 0);
/// ```
pub fn synthesize_from_config(
    config: &ServerConfig,
) -> Result<Vec<(String, ToolInfo, Arc<dyn ToolHandler>)>> {
    let mut out = Vec::with_capacity(config.tools.len());
    for decl in &config.tools {
        let schema = build_input_schema(&decl.parameters);
        let annotations = build_annotations(&decl.annotations);
        let info = if let Some(ann) = annotations {
            ToolInfo::with_annotations(
                decl.name.clone(),
                Some(decl.description.clone()),
                schema,
                ann,
            )
        } else {
            ToolInfo::new(
                decl.name.clone(),
                Some(decl.description.clone()),
                schema,
            )
        };
        let handler: Arc<dyn ToolHandler> = Arc::new(SynthesizedToolHandler {
            info: info.clone(),
            decl: decl.clone(),
        });
        out.push((decl.name.clone(), info, handler));
    }
    Ok(out)
}

/// Build a JSON Schema `properties` + `required` object from `[[tools.parameters]]`.
///
/// Decomposed from `synthesize_from_config` to keep cognitive complexity ≤25
/// (CLAUDE.md Phase 75 D-03 + PATTERNS Pattern G).
fn build_input_schema(params: &[ParamDecl]) -> Value {
    let mut props = Map::new();
    let mut required = Vec::new();
    for p in params {
        let prop = build_param_property(p);
        props.insert(p.name.clone(), prop);
        if p.required {
            required.push(Value::String(p.name.clone()));
        }
    }
    json!({
        "type": "object",
        "properties": props,
        "required": required,
        "additionalProperties": false,
    })
}

/// Build one JSON Schema property from a single ParamDecl.
fn build_param_property(p: &ParamDecl) -> Value {
    let mut prop = json!({
        "type": p.json_schema_type(),
        "description": p.description,
    });
    if let Some(min) = p.min {
        prop["minimum"] = json!(min);
    }
    if let Some(max) = p.max {
        prop["maximum"] = json!(max);
    }
    if let Some(max_len) = p.max_length {
        prop["maxLength"] = json!(max_len);
    }
    if let Some(default) = &p.default {
        prop["default"] = default.clone();
    }
    prop
}

/// Build `ToolAnnotations` from an optional `[tools.annotations]` block.
fn build_annotations(decl: &Option<AnnotationsDecl>) -> Option<ToolAnnotations> {
    let d = decl.as_ref()?;
    let mut a = ToolAnnotations::new();
    if let Some(v) = d.read_only_hint {
        a = a.with_read_only(v);
    }
    if let Some(v) = d.destructive_hint {
        a = a.with_destructive(v);
    }
    if let Some(v) = d.idempotent_hint {
        a = a.with_idempotent(v);
    }
    if let Some(v) = d.open_world_hint {
        a = a.with_open_world(v);
    }
    Some(a)
}
