/// Resolve `token_secret` config value into a `TokenSecret`.
///
/// Accepts three shapes:
/// - `Some("env:VAR_NAME")` → looks up `VAR_NAME` via `crate::secrets::EnvSecrets`
/// - `Some("<inline-secret>")` → treats as raw secret (dev/test only; warn via tracing)
/// - `None` → returns `Err(ToolkitError::CodeMode("token_secret required"))`
fn resolve_token_secret(section: &CodeModeSection) -> Result<TokenSecret> {
    let raw = section.token_secret.as_ref().ok_or_else(|| {
        ToolkitError::CodeMode("[code_mode] token_secret is required when code-mode is enabled".to_string())
    })?;
    if let Some(var) = raw.strip_prefix("env:") {
        std::env::var(var)
            .map(|s| TokenSecret::new(s.into_bytes()))
            .map_err(|_| ToolkitError::CodeMode(format!("env var '{}' not set for token_secret", var)))
    } else {
        // Inline secret — warn (dev/test pattern)
        tracing::warn!(
            "[code_mode] token_secret is inline; prefer 'env:VAR_NAME' for production deployments"
        );
        Ok(TokenSecret::new(raw.as_bytes().to_vec()))
    }
}
