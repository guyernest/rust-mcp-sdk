#[async_trait]
impl PromptHandler for SimplePrompt {
    async fn handle(&self, args: HashMap<String, String>, _extra: RequestHandlerExtra)
        -> Result<GetPromptResult> {
        for arg in &self.arguments {
            if arg.required && !args.contains_key(&arg.name) {
                return Err(pmcp::Error::validation(format!(
                    "Required argument '{}' is missing", arg.name
                )));
            }
        }
        // ... render
    }
    fn metadata(&self) -> Option<PromptInfo> {
        let mut info = PromptInfo::new(&self.name);
        if let Some(desc) = &self.description { info = info.with_description(desc); }
        if !self.arguments.is_empty() { info = info.with_arguments(self.arguments.clone()); }
        Some(info)
    }
}
