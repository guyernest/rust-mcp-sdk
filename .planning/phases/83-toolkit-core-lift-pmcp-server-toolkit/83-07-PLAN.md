/// Per spike 005 §"What to Avoid", `translate_placeholders` is intentionally

#[async_trait]
pub trait SqlConnector: Send + Sync + 'static {
    /// Identify the dialect for placeholder translation and prompt assembly.
    fn dialect(&self) -> Dialect;

    /// Execute a query with named placeholders (`:name` form) and named params.
    ///
    /// Implementations apply [`translate_placeholders`] to convert the canonical
    /// `:name` form into their dialect's native placeholder shape before driver
    /// execution. Returns a `Vec<Value>` where each `Value` is one row (typically
    /// an object).
    async fn execute(
        &self,
        sql_with_named_placeholders: &str,
        named_params: &[(String, Value)],
    ) -> Result<Vec<Value>, ConnectorError>;

    /// Render the backend's schema as DDL or equivalent text for inclusion in
    /// the code-mode prompt. Phase 84 impls drive this from `information_schema`,
    /// Glue catalog, or `sqlite_master` per dialect.
    async fn schema_text(&self) -> Result<String, ConnectorError>;
}

/// Supported SQL dialects (4-variant per spike 005).
///
/// `#[non_exhaustive]` permits additive evolution to Dialect::Oracle / ::SqlServer /
/// ::DuckDb / ::ClickHouse in later phases without semver break.
#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]

pub enum Dialect {
    Postgres,
    MySql,
    Athena,
    Sqlite,
}

impl Dialect {
    /// Stable, human-readable name for prompts and logs.
    pub const fn name(self) -> &'static str {
        match self {
            Self::Postgres => "PostgreSQL",
            Self::MySql => "MySQL",
            Self::Athena => "Amazon Athena (Presto/Trino)",
            Self::Sqlite => "SQLite",
        }
    }

    /// One-line guidance string for the code-mode prompt body explaining the
    /// dialect's placeholder convention.
    pub const fn placeholder_guidance(self) -> &'static str {
        match self {
            Self::Postgres => "Use $1, $2, $3, ... for positional parameters.",
            Self::MySql => "Use ? for positional parameters in argument order.",
            Self::Athena => "Use ? for positional parameters in argument order.",
            Self::Sqlite => "Use :name for named parameters or ? for positional.",
        }
    }
}

/// Errors a `SqlConnector` impl may surface.
#[derive(Debug, Error)]
#[non_exhaustive]
pub enum ConnectorError {
    #[error("connector I/O error: {0}")]
    Io(String),

    #[error("schema fetch failed: {0}")]
    Schema(String),

    #[error("query execution failed: {0}")]
    Query(String),

    #[error("dialect mismatch: query used {used:?} but connector is {actual:?}")]
    DialectMismatch { used: Dialect, actual: Dialect },
}

/// Translate `:name` named placeholders to the dialect's native form.
///
/// Per spike 005 §"What to Avoid": this is a FREE FUNCTION, not a trait method.
/// Per-backend impls of `SqlConnector::execute` call this helper before passing
/// SQL to their driver.
///
/// # Dialect rules
///
/// - **Postgres:** `:name` → `$1`, `$2`, ... in order of first appearance.
/// - **MySQL / Athena:** `:name` → `?` in order of first appearance.
/// - **SQLite:** `:name` → `:name` (unchanged; SQLite natively supports named).
///
/// # Example
///
/// ```
/// use pmcp_server_toolkit::sql::{Dialect, translate_placeholders};
///
/// let sql = "SELECT * FROM t WHERE a = :x AND b = :y";
/// assert_eq!(translate_placeholders(Dialect::Postgres, sql),
///            "SELECT * FROM t WHERE a = $1 AND b = $2");
/// assert_eq!(translate_placeholders(Dialect::MySql, sql),
///            "SELECT * FROM t WHERE a = ? AND b = ?");
/// assert_eq!(translate_placeholders(Dialect::Sqlite, sql), sql);
/// ```
pub fn translate_placeholders(dialect: Dialect, sql: &str) -> String {
    match dialect {
        Dialect::Sqlite => sql.to_string(),
        Dialect::Postgres => translate_to_positional(sql, |i| format!("${}", i + 1)),
        Dialect::MySql | Dialect::Athena => translate_to_positional(sql, |_| "?".to_string()),
    }
}

/// Internal helper: walk `:name` placeholders in order and replace via `replacement(idx)`.
///
/// Conservative parser — does not look inside string literals. Sufficient for
/// the curated [[tools]] SQL shape (which is operator-authored and trusted).
/// Phase 84 may upgrade to a full SQL tokenizer if real-world configs surface
/// edge cases.
fn translate_to_positional<F: FnMut(usize) -> String>(sql: &str, mut replacement: F) -> String {
    let mut out = String::with_capacity(sql.len());
    let mut chars = sql.chars().peekable();
    let mut idx = 0usize;
    while let Some(ch) = chars.next() {
        if ch == ':' && chars.peek().map(|c| c.is_ascii_alphabetic() || *c == '_').unwrap_or(false) {
            // Consume the identifier (alnum + underscore)
            while let Some(&c) = chars.peek() {
                if c.is_ascii_alphanumeric() || c == '_' {
                    chars.next();
                } else {
                    break;
                }
            }
            out.push_str(&replacement(idx));
            idx += 1;
        } else {
            out.push(ch);
        }
    }
    out
}

use crate::config::ServerConfig;
use crate::sql::{Dialect, SqlConnector};

/// TKIT-10: assemble the code-mode bootstrap prompt body from a connector's
/// schema text + curated `[[database.tables]]` descriptions.
///
/// Output structure:
///
/// ```text
/// # Code Mode — {dialect.name()}
///
/// {dialect.placeholder_guidance()}
///
/// ## Schema
///
/// {connector.schema_text()}
///
/// ## Curated Tables
///
/// - `table_a`: description A
/// - `table_b`: description B
/// ```
///
/// If `[[database.tables]]` is empty, the "Curated Tables" section is omitted.
///
/// # Errors
///
/// Returns `crate::error::ToolkitError::CodeMode` if `connector.schema_text()`
/// fails. The toolkit does not retry; callers should ensure the connector is
/// ready before assembling.
///
/// # Example
///
/// ```no_run
/// # use pmcp_server_toolkit::config::ServerConfig;
/// # use pmcp_server_toolkit::sql::{Dialect, SqlConnector};
/// # use pmcp_server_toolkit::code_mode::assemble_code_mode_prompt;
/// # async fn example(connector: &dyn SqlConnector, config: &ServerConfig) {
/// let prompt = assemble_code_mode_prompt(connector, config).await.unwrap();
/// assert!(prompt.contains("# Code Mode"));
/// # }
/// ```
pub async fn assemble_code_mode_prompt(
    connector: &dyn SqlConnector,
    config: &ServerConfig,
) -> Result<String> {
    let dialect = connector.dialect();
    let schema_text = connector
        .schema_text()
        .await
        .map_err(|e| ToolkitError::CodeMode(format!("schema_text failed: {}", e)))?;

    let curated = format_curated_tables(config);

    let mut out = String::with_capacity(schema_text.len() + curated.len() + 256);
    out.push_str(&format!("# Code Mode — {}\n\n", dialect.name()));
    out.push_str(dialect.placeholder_guidance());
    out.push_str("\n\n## Schema\n\n");
    out.push_str(&schema_text);
    if !curated.is_empty() {
        out.push_str("\n\n## Curated Tables\n\n");
        out.push_str(&curated);
    }
    out.push('\n');
    Ok(out)
}

/// Format the `[[database.tables]]` curated descriptions as a Markdown list.
fn format_curated_tables(config: &ServerConfig) -> String {
    config
        .database
        .tables
        .iter()
        .map(|t| format!("- `{}`: {}", t.name, t.description))
        .collect::<Vec<_>>()
        .join("\n")
}
