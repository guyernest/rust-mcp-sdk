//! Builder extension trait for `pmcp::ServerBuilder` — connects config-driven
//! synthesis (Plans 04, 05, 06) to the public Phase 82 builder API.
//!
//! Per CONTEXT.md D-10 + D-11, this is the "common path" surface — power users
//! call `tools::synthesize_from_config` + `code_mode::register_code_mode_tools`
//! directly. Shape C ≤15-line `main.rs` users compose this trait.
//!
//! Per review R7, each method has a panicking convenience form (`tools_from_config`,
//! `code_mode_from_config`) AND a fallible companion (`try_tools_from_config`,
//! `try_code_mode_from_config`). The panicking forms delegate to the `try_*`
//! variants with documented panic messages — production servers should prefer
//! the `try_*` shape so misconfiguration surfaces as a Result, not a crash.

use pmcp::ServerBuilder;

use crate::config::ServerConfig;
use crate::error::Result;

/// Composable builder extensions for config-driven `pmcp` servers.
pub trait ServerBuilderExt: Sized {
    /// Register every `[[tools]]` entry from `config` as a `tool_arc` handler
    /// (TKIT-07). Panicking convenience wrapping `try_tools_from_config`.
    ///
    /// # Panics
    ///
    /// Panics with `"tools_from_config: ..."` if `try_tools_from_config` errors
    /// (e.g. synthesis fails). Prefer `try_tools_from_config` for production
    /// servers where misconfiguration must surface as a `Result`.
    ///
    /// # Example
    ///
    /// ```no_run
    /// # use pmcp::Server;
    /// # use pmcp_server_toolkit::{ServerBuilderExt, ServerConfig};
    /// # let cfg = ServerConfig::default();
    /// let builder = Server::builder()
    ///     .name("demo").version("0.1.0")
    ///     .tools_from_config(&cfg);
    /// ```
    fn tools_from_config(self, config: &ServerConfig) -> Self;

    /// Fallible companion to `tools_from_config` (review R7).
    ///
    /// # Errors
    ///
    /// Returns `ToolkitError` if synthesis fails (typically `ToolkitError::Synth`
    /// or `ToolkitError::Validation`).
    ///
    /// # Example
    ///
    /// ```no_run
    /// # use pmcp::Server;
    /// # use pmcp_server_toolkit::{ServerBuilderExt, ServerConfig};
    /// # let cfg = ServerConfig::default();
    /// # fn run() -> Result<(), Box<dyn std::error::Error>> {
    /// let builder = Server::builder()
    ///     .name("demo").version("0.1.0")
    ///     .try_tools_from_config(&cfg)?;
    /// # Ok(()) }
    /// ```
    fn try_tools_from_config(self, config: &ServerConfig) -> Result<Self>;

    /// Wire the `[code_mode]` block. Panicking convenience wrapping
    /// `try_code_mode_from_config`.
    ///
    /// When the `code-mode` feature is disabled, this is a no-op.
    ///
    /// # Panics
    ///
    /// Panics if `try_code_mode_from_config` errors (e.g. `token_secret`
    /// env var unset; inline literal without dev flag per review R9). Prefer
    /// `try_code_mode_from_config` for production servers.
    ///
    /// # Example
    ///
    /// ```no_run
    /// # use pmcp::Server;
    /// # use pmcp_server_toolkit::{ServerBuilderExt, ServerConfig};
    /// # let cfg = ServerConfig::default();
    /// let builder = Server::builder()
    ///     .name("demo").version("0.1.0")
    ///     .code_mode_from_config(&cfg);
    /// ```
    fn code_mode_from_config(self, config: &ServerConfig) -> Self;

    /// Fallible companion to `code_mode_from_config` (review R7).
    ///
    /// # Errors
    ///
    /// Returns `ToolkitError` if code-mode wiring fails (`ToolkitError::CodeMode`,
    /// `ToolkitError::Validation`).
    ///
    /// # Example
    ///
    /// ```no_run
    /// # use pmcp::Server;
    /// # use pmcp_server_toolkit::{ServerBuilderExt, ServerConfig};
    /// # let cfg = ServerConfig::default();
    /// # fn run() -> Result<(), Box<dyn std::error::Error>> {
    /// let builder = Server::builder()
    ///     .name("demo").version("0.1.0")
    ///     .try_code_mode_from_config(&cfg)?;
    /// # Ok(()) }
    /// ```
    fn try_code_mode_from_config(self, config: &ServerConfig) -> Result<Self>;
}

impl ServerBuilderExt for ServerBuilder {
    fn tools_from_config(self, config: &ServerConfig) -> Self {
        self.try_tools_from_config(config)
            .expect("tools_from_config: synthesize_from_config returned an error — \
                     prefer try_tools_from_config to handle this as a Result")
    }

    fn try_tools_from_config(mut self, config: &ServerConfig) -> Result<Self> {
        let synthesized = crate::tools::synthesize_from_config(config)?;
        for (name, _info, handler) in synthesized {
            self = self.tool_arc(name, handler);
        }
        Ok(self)
    }

    fn code_mode_from_config(self, config: &ServerConfig) -> Self {
        self.try_code_mode_from_config(config)
            .expect("code_mode_from_config: register_code_mode_tools errored — \
                     prefer try_code_mode_from_config to handle (e.g. missing env var)")
    }

    fn try_code_mode_from_config(self, config: &ServerConfig) -> Result<Self> {
        #[cfg(feature = "code-mode")]
        {
            return crate::code_mode::register_code_mode_tools(self, config);
        }
        #[cfg(not(feature = "code-mode"))]
        {
            let _ = config;
            tracing::warn!(
                "try_code_mode_from_config called but `code-mode` feature is disabled; skipping"
            );
            Ok(self)
        }
    }
}
