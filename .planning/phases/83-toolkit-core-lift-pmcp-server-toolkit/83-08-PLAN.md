//! Builder extension trait for `pmcp::ServerBuilder` — connects config-driven
//! synthesis (Plans 04, 05, 06) to the public Phase 82 builder API.
//!
//! Per CONTEXT.md D-10 + D-11, this is the "common path" surface — power users
//! call `tools::synthesize_from_config` + `code_mode::register_code_mode_tools`
//! directly. Shape C ≤15-line `main.rs` users compose this trait.

use pmcp::ServerBuilder;

use crate::config::ServerConfig;

/// Composable builder extensions for config-driven `pmcp` servers.
///
/// Wired to `pmcp::ServerBuilder` so any `Server::builder()` chain can pull in
/// `[[tools]]` synthesis and `[code_mode]` wiring with one method call each.
pub trait ServerBuilderExt: Sized {
    /// Register every `[[tools]]` entry from `config` as a `tool_arc` handler
    /// (TKIT-07).
    ///
    /// # Panics
    ///
    /// Panics if `synthesize_from_config` returns an error. `ServerConfig` is
    /// parse-time-invariant per D-13; an error here indicates an internal bug.
    /// If you need fallible registration, call
    /// [`crate::tools::synthesize_from_config`] directly and handle the result.
    ///
    /// # Example
    ///
    /// ```no_run
    /// # use pmcp::Server;
    /// # use pmcp_server_toolkit::{ServerBuilderExt, config::ServerConfig};
    /// # let cfg = ServerConfig::default();
    /// let builder = Server::builder()
    ///     .name("demo").version("0.1.0")
    ///     .tools_from_config(&cfg);
    /// ```
    fn tools_from_config(self, config: &ServerConfig) -> Self;

    /// Wire the `[code_mode]` block into `pmcp_code_mode::CodeExecutor` and
    /// register the `validate_code` + `execute_code` tools (TKIT-09).
    ///
    /// When the `code-mode` feature is disabled, this is a no-op.
    ///
    /// # Panics
    ///
    /// Panics if `register_code_mode_tools` returns an error (e.g. `token_secret`
    /// env var unset). For fallible wiring, call
    /// [`crate::code_mode::register_code_mode_tools`] directly.
    ///
    /// # Example
    ///
    /// ```no_run
    /// # use pmcp::Server;
    /// # use pmcp_server_toolkit::{ServerBuilderExt, config::ServerConfig};
    /// # let cfg = ServerConfig::default();
    /// let builder = Server::builder()
    ///     .name("demo").version("0.1.0")
    ///     .code_mode_from_config(&cfg);
    /// ```
    fn code_mode_from_config(self, config: &ServerConfig) -> Self;
}

impl ServerBuilderExt for ServerBuilder {
    fn tools_from_config(mut self, config: &ServerConfig) -> Self {
        let synthesized = crate::tools::synthesize_from_config(config)
            .expect("synthesize_from_config: ServerConfig is parse-time-invariant per D-13");
        for (name, _info, handler) in synthesized {
            self = self.tool_arc(name, handler);
        }
        self
    }

    fn code_mode_from_config(self, config: &ServerConfig) -> Self {
        #[cfg(feature = "code-mode")]
        {
            return crate::code_mode::register_code_mode_tools(self, config)
                .expect("register_code_mode_tools: see error for token_secret / config issues");
        }
        #[cfg(not(feature = "code-mode"))]
        {
            // No-op when code-mode feature is disabled. Document via tracing for
            // observability — users may not realize the feature is off.
            let _ = config;
            tracing::warn!(
                "code_mode_from_config called but `code-mode` feature is disabled; skipping"
            );
            self
        }
    }
}
