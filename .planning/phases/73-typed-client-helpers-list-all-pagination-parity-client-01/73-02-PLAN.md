/// List all tools across all pages, auto-paginating on `next_cursor`.
///
/// Loops calling [`Self::list_tools`], terminating when the server returns
/// `next_cursor: None`. Safety cap: if the loop runs more than
/// `self.options.max_iterations` iterations (default `100`), returns
/// [`Error::validation`] instead of continuing or silently truncating.
///
/// # Errors
///
/// - Any error surfaced by [`Self::list_tools`] propagates unchanged.
/// - Cap exceeded → `Error::Validation("list_all_tools exceeded max_iterations cap of N pages")`.
///
/// # Examples
///
/// ```rust,no_run
/// # async fn ex<T: pmcp::shared::Transport + Send + Sync + 'static>(mut client: pmcp::Client<T>) -> pmcp::Result<()> {
/// let tools = client.list_all_tools().await?;
/// println!("discovered {} tools", tools.len());
/// # Ok(()) }
/// ```
pub async fn list_all_tools(&self) -> Result<Vec<ToolInfo>> {
    let cap = self.options.max_iterations;
    let mut out: Vec<ToolInfo> = Vec::new();
    let mut cursor: Option<String> = None;
    for _ in 0..cap {
        let page = self.list_tools(cursor).await?;
        out.extend(page.tools);
        match page.next_cursor {
            None => return Ok(out),
            Some(next) => cursor = Some(next),
        }
    }
    Err(Error::validation(format!(
        "list_all_tools exceeded max_iterations cap of {cap} pages"
    )))
}

```rust
//! Integration tests for `Client::list_all_*` auto-pagination (Phase 73,
//! PARITY-CLIENT-01). Drives a MockTransport with scripted multi-page
//! response sequences and asserts:
//!   1. Aggregation across pages preserves order.
//!   2. Termination on `next_cursor: None`.
//!   3. `max_iterations` cap enforcement returns `Error::Validation`.

#![cfg(not(target_arch = "wasm32"))]

use async_trait::async_trait;
use pmcp::{
    shared::Transport,
    types::{
        jsonrpc::ResponsePayload, ClientCapabilities, JSONRPCResponse, RequestId,
        TransportMessage,
    },
    Client, ClientOptions, Error, Result,
};
use serde_json::json;
use std::sync::{Arc, Mutex};

// --- MockTransport: inlined copy of src/client/mod.rs:1847-1891 ---

#[derive(Debug)]
struct MockTransport {
    responses: Arc<Mutex<Vec<TransportMessage>>>,
    sent_messages: Arc<Mutex<Vec<TransportMessage>>>,
}

impl MockTransport {
    fn with_responses(responses: Vec<TransportMessage>) -> Self {
        Self {
            responses: Arc::new(Mutex::new(responses)),
            sent_messages: Arc::new(Mutex::new(Vec::new())),
        }
    }
}

#[async_trait]
impl Transport for MockTransport {
    async fn send(&mut self, message: TransportMessage) -> Result<()> {
        self.sent_messages.lock().unwrap().push(message);
        Ok(())
    }
    async fn receive(&mut self) -> Result<TransportMessage> {
        self.responses
            .lock()
            .unwrap()
            .pop()
            .ok_or_else(|| pmcp::Error::protocol_msg("no more responses"))
    }
    async fn close(&mut self) -> Result<()> {
        Ok(())
    }
}

// --- Response builders ---

fn init_response() -> TransportMessage {
    TransportMessage::Response(JSONRPCResponse {
        jsonrpc: "2.0".to_string(),
        id: RequestId::from(1i64),
        payload: ResponsePayload::Result(json!({
            "protocolVersion": "2025-06-18",
            "capabilities": { "tools": {} },
            "serverInfo": { "name": "test-server", "version": "1.0.0" }
        })),
    })
}

fn tools_page(id: i64, names: &[&str], next: Option<&str>) -> TransportMessage {
    let tools: Vec<_> = names
        .iter()
        .map(|n| json!({ "name": n, "description": "t", "inputSchema": {} }))
        .collect();
    let mut payload = json!({ "tools": tools });
    if let Some(cur) = next {
        payload["nextCursor"] = json!(cur);
    }
    TransportMessage::Response(JSONRPCResponse {
        jsonrpc: "2.0".to_string(),
        id: RequestId::from(id),
        payload: ResponsePayload::Result(payload),
    })
}
```

```rust
//! Fuzz target: Client::list_all_tools cursor loop (Phase 73, T-73-01).
//!
//! Invariants:
//!   1. Loop terminates within ClientOptions::max_iterations for any
//!      adversarial cursor sequence (empty strings, very long strings,
//!      repeated values, A→B→A cycles).
//!   2. Cap-exceeded yields `Error::Validation`, never panics.
//!   3. None-cursor terminates cleanly with `Ok(accumulator)`.
//!   4. No `Error` variant other than `Validation` or delegated
//!      transport errors surfaces.

#![no_main]

use arbitrary::{Arbitrary, Unstructured};
use async_trait::async_trait;
use libfuzzer_sys::fuzz_target;
use pmcp::{
    shared::Transport,
    types::{
        jsonrpc::ResponsePayload, ClientCapabilities, JSONRPCResponse, RequestId,
        TransportMessage,
    },
    Client, ClientOptions, Error,
};
use serde_json::json;
use std::sync::{Arc, Mutex};

#[derive(Debug, Arbitrary)]
struct FuzzCursorSeq {
    /// Each entry: `Some(next_cursor)` continues, `None` terminates.
    /// Empty strings, long strings, repeated values are all valid.
    cursors: Vec<Option<String>>,
    /// Clamped 1..=200 inside the target to keep runs bounded.
    max_iterations: u8,
}

#[derive(Debug)]
struct MockTransport {
    responses: Arc<Mutex<Vec<TransportMessage>>>,
}

#[async_trait]
impl Transport for MockTransport {
    async fn send(&mut self, _: TransportMessage) -> pmcp::Result<()> {
        Ok(())
    }
    async fn receive(&mut self) -> pmcp::Result<TransportMessage> {
        self.responses
            .lock()
            .unwrap()
            .pop()
            .ok_or_else(|| Error::protocol_msg("no more responses"))
    }
    async fn close(&mut self) -> pmcp::Result<()> {
        Ok(())
    }
}

fn build_responses(seq: &FuzzCursorSeq) -> Vec<TransportMessage> {
    // init response comes LAST so pop() yields it first.
    let init = TransportMessage::Response(JSONRPCResponse {
        jsonrpc: "2.0".to_string(),
        id: RequestId::from(1i64),
        payload: ResponsePayload::Result(json!({
            "protocolVersion": "2025-06-18",
            "capabilities": { "tools": {} },
            "serverInfo": { "name": "fuzz", "version": "0.0.0" }
        })),
    });

    // Page responses: each element of seq.cursors becomes one page.
    // The `.cursors` vec is iterated in order; we need to push in reverse
    // so the FIRST page pops first.
    let mut pages: Vec<TransportMessage> = seq
        .cursors
        .iter()
        .enumerate()
        .map(|(i, cur)| {
            let mut payload = json!({ "tools": [] });
            if let Some(next) = cur {
                payload["nextCursor"] = json!(next);
            }
            TransportMessage::Response(JSONRPCResponse {
                jsonrpc: "2.0".to_string(),
                id: RequestId::from((i as i64) + 2),
                payload: ResponsePayload::Result(payload),
            })
        })
        .collect();
    pages.reverse();
    pages.push(init);
    pages
}

fuzz_target!(|data: &[u8]| {
    let mut u = Unstructured::new(data);
    let Ok(seq) = FuzzCursorSeq::arbitrary(&mut u) else {
        return;
    };
    let cap = (seq.max_iterations as usize).clamp(1, 200);

    let responses = build_responses(&seq);
    let transport = MockTransport {
        responses: Arc::new(Mutex::new(responses)),
    };
    let opts = ClientOptions {
        max_iterations: cap,
        ..Default::default()
    };

    let rt = tokio::runtime::Runtime::new().expect("runtime");
    rt.block_on(async move {
        let mut client = Client::with_client_options(transport, opts);
        let _ = client.initialize(ClientCapabilities::minimal()).await;
        let outcome = client.list_all_tools().await;
        match outcome {
            Ok(_) => {}
            // Validation is the expected error variant for cap-exceeded.
            Err(Error::Validation(_)) => {}
            // Transport-layer protocol errors are acceptable (occur when
            // the fuzz scripted fewer pages than needed). Anything else
            // is a bug.
            Err(Error::Protocol { .. }) | Err(Error::Parse(_)) => {}
            Err(other) => panic!("unexpected error variant: {other:?}"),
        }
    });
});
```

| Boundary | Description |

| server → `Client::list_all_*` cursor loop | Server-controlled `next_cursor` value is read and re-sent in the next request. An untrusted or buggy server could emit a never-terminating cursor chain. |
| adversarial fuzzer → `Client::list_all_tools` | `cargo fuzz` driver feeds arbitrary cursor sequences to exercise the cap + termination invariants. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| **T-73-01** | **Denial of Service** | `Client::list_all_*` cursor loop | **mitigate** | Hard `max_iterations` cap (default 100) enforced by bounded `for _ in 0..cap` loop. Cap-exceeded returns `Error::validation(format!("list_all_* exceeded max_iterations cap of {cap} pages"))`. Property test `prop_list_all_tools_cap_enforced` randomizes `cap in 1..20` and confirms cap+1 `Some(_)` pages always error out. Fuzz target `list_all_cursor_loop` exercises empty / long / repeated / cyclic cursors under adversarial `max_iterations` values (clamped 1..=200). |
| T-73-02 | Tampering | Server emits a cursor that cycles back (A→B→A→B…) | mitigate | Bounded-iteration cap from T-73-01 terminates regardless of cursor semantics. Cycle detection is intentionally NOT implemented — MCP spec does not require cursor uniqueness (per RESEARCH §Security Domain). |
| T-73-03 | Information Disclosure | Very-long-string cursor leaks server state via client memory pressure | accept | `cursor: Option<String>` is re-sent verbatim; this matches the existing `list_tools` semantics and is not new attack surface. Client owns the cap knob if memory pressure becomes a concern. |
| T-73-04 | Denial of Service | Property test generates `max_iterations = 0` (ClientOptions constructed with degenerate cap) | accept | `for _ in 0..0` is a no-op; the loop immediately falls through to the `Err(Error::validation(...))` branch. No panic, no hang. Behavior documented in rustdoc. |
