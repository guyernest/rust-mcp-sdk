```javascript
        // Setup DevTools toggle button (D-01)
        const toggleBtn = document.getElementById('devtools-toggle');
        const devtoolsPanel = document.querySelector('.devtools-panel');
        toggleBtn.addEventListener('click', () => {
          if (devtoolsPanel.classList.contains('collapsed')) {
            devtoolsPanel.style.width = '350px';
            devtoolsPanel.classList.remove('collapsed');
            toggleBtn.classList.add('panel-open');

            toggleBtn.classList.remove('panel-open');
          }
        });
```

Add resize handle handler:
```javascript
        // Setup resize handle (D-02, D-03)
        const resizeHandle = document.getElementById('devtools-resize-handle');
        resizeHandle.addEventListener('mousedown', (e) => {
          e.preventDefault();
          const startX = e.clientX;
          const startWidth = devtoolsPanel.getBoundingClientRect().width;
          devtoolsPanel.classList.add('resizing');
          resizeHandle.classList.add('dragging');
          document.body.style.userSelect = 'none';
          document.body.style.cursor = 'col-resize';

          const onMouseMove = (e) => {
            const delta = startX - e.clientX;
            const maxWidth = window.innerWidth * 0.8;
            const newWidth = Math.max(0, Math.min(maxWidth, startWidth + delta));
            devtoolsPanel.style.width = newWidth + 'px';
            if (newWidth === 0) {
              devtoolsPanel.classList.add('collapsed');
              toggleBtn.classList.remove('panel-open');
            } else {
              devtoolsPanel.classList.remove('collapsed');
              toggleBtn.classList.add('panel-open');
            }
          };

          const onMouseUp = () => {
            document.removeEventListener('mousemove', onMouseMove);
            document.removeEventListener('mouseup', onMouseUp);
            devtoolsPanel.classList.remove('resizing');
            resizeHandle.classList.remove('dragging');
            document.body.style.userSelect = '';
            document.body.style.cursor = '';
          };

          document.addEventListener('mousemove', onMouseMove);
          document.addEventListener('mouseup', onMouseUp);
        });
```

```javascript
        // Setup global Clear All button (D-07)

          document.querySelectorAll('.clear-btn').forEach(btn => btn.click());
        });
```
