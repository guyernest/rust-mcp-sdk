   - Backoff: 500ms → 1000ms → 2000ms → 4000ms → 5000ms (capped); total bounded by `query_timeout_ms`
   - Convert `ResultSet` rows → `Vec<Value::Object>` via column-header projection from row 0

   - For each table name in `self.tables` (populated via `with_tables(...)`), call `client.get_table_metadata(catalog="AwsDataCatalog", database, table_name).send()`
   - Assemble CREATE-EXTERNAL-TABLE blob with Columns + PartitionKeys + Parameters per RESEARCH §1.3

        pub tables: std::collections::HashMap<String, Vec<serde_json::Value>>,
        pub last_translated_sql: std::sync::Mutex<Option<String>>,
        pub last_positional_args: std::sync::Mutex<Option<Vec<serde_json::Value>>>,
    }
    
    impl AthenaMock {
        pub fn open_images_fixture() -> Self {
            let mut tables = std::collections::HashMap::new();
            tables.insert("images".into(), vec![
                serde_json::json!({"image_id": "img1", "label": "cat"}),
                serde_json::json!({"image_id": "img2", "label": "dog"}),
            ]);
            Self {
                tables,
                last_translated_sql: std::sync::Mutex::new(None),
                last_positional_args: std::sync::Mutex::new(None),
            }
        }
    }
    
    #[async_trait::async_trait]
    impl pmcp_server_toolkit::sql::SqlConnector for AthenaMock {
        fn dialect(&self) -> pmcp_server_toolkit::sql::Dialect {
            pmcp_server_toolkit::sql::Dialect::Athena
        }
    
        async fn execute(&self, sql: &str, params: &[(String, serde_json::Value)])
            -> Result<Vec<serde_json::Value>, pmcp_server_toolkit::sql::ConnectorError>
        {
            let pmcp_server_toolkit::sql::TranslatedSql { sql: translated, ordered_params } =
                pmcp_server_toolkit::sql::translate_placeholders(sql, pmcp_server_toolkit::sql::Dialect::Athena);
            let positional: Vec<serde_json::Value> = ordered_params.iter()
                .map(|n| params.iter().find(|(k,_)| k == n).map(|(_,v)| v.clone()).unwrap_or(serde_json::Value::Null))
                .collect();
            *self.last_translated_sql.lock().unwrap() = Some(translated.clone());
            *self.last_positional_args.lock().unwrap() = Some(positional.clone());
            cheap_query_engine(&self.tables, &translated, &positional)
        }
    
        async fn schema_text(&self) -> Result<String, pmcp_server_toolkit::sql::ConnectorError> {
            // Glue-Data-Catalog-styled CREATE EXTERNAL TABLE.
            Ok(
                "CREATE EXTERNAL TABLE images (\n  image_id STRING,\n  label STRING\n)\n\
                 PARTITIONED BY (\n  dt STRING\n)\n\
                 LOCATION 's3://example-bucket/images/';\n".to_string()
            )
        }
    }
    
    fn cheap_query_engine(
        tables: &std::collections::HashMap<String, Vec<serde_json::Value>>,
        sql: &str,
        args: &[serde_json::Value],
    ) -> Result<Vec<serde_json::Value>, pmcp_server_toolkit::sql::ConnectorError> {
        if sql.contains("FROM images WHERE image_id = ?") {
            let id = args.first().and_then(|v| v.as_str()).unwrap_or("");
            return Ok(
                tables.get("images").cloned().unwrap_or_default()
                    .into_iter()
                    .filter(|r| r["image_id"].as_str() == Some(id))
                    .collect()
            );
        }
        if sql.contains("SELECT * FROM images") {
            return Ok(tables.get("images").cloned().unwrap_or_default());
        }
        Ok(vec![])
    }
