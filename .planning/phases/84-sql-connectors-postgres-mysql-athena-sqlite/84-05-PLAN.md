        fn dialect(&self) -> Dialect { Dialect::Postgres }
    
        async fn execute(&self, sql: &str, params: &[(String, Value)]) -> Result<Vec<Value>, ConnectorError> {
            let TranslatedSql { sql: translated, ordered_params } =
                translate_placeholders(sql, Dialect::Postgres);
            let owned: Vec<PgParam> = ordered_params.iter()
                .map(|n| params.iter().find(|(k,_)| k == n)
                    .map(|(_,v)| value_to_pg_param(v))
                    .unwrap_or(PgParam::Null))
                .collect();
            let refs: Vec<&(dyn ToSql + Sync)> = owned.iter().map(|p| p as &(dyn ToSql + Sync)).collect();

                .map_err(|e| ConnectorError::Connection(format!("pool acquire: {e}")))?;
            let rows = client.query(&translated, &refs).await
                .map_err(|e| ConnectorError::Query(e.to_string()))?;
            Ok(rows.iter().map(row_to_value).collect())
