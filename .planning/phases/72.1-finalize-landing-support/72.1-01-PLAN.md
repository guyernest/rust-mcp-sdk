```typescript
'use client'
import { useEffect, useState } from 'react'

export type LandingConfig = {
  /** Public MCP endpoint — always present. */
  mcpEndpoint: string
  /** Sign-up config, or null when OAuth is not enabled / landing client missing. */
  signup: {
    cognitoDomain: string
    cognitoRegion: string
    landingClientId: string
    signupRedirectAfter: string
  } | null
}

/**
 * Resolved states:
 *   null      — still loading
 *   'error'   — fetch failed / 404 / unexpected shape
 *   LandingConfig — successful load (signup may itself be null when disabled)
 */
export type LandingConfigState = LandingConfig | null | 'error'

export function useLandingConfig(): LandingConfigState {
  const [cfg, setCfg] = useState<LandingConfigState>(null)
  useEffect(() => {
    let cancelled = false
    fetch('/landing-config', { cache: 'no-store' })
      .then((r) => (r.ok ? r.json() : Promise.reject(r.status)))
      .then((c) => { if (!cancelled) setCfg(c as LandingConfig) })
      .catch(() => { if (!cancelled) setCfg('error') })
    return () => { cancelled = true }
  }, [])
  return cfg
}
```

**2a. `cargo-pmcp/templates/landing/nextjs/app/signup/page.tsx`** — full replacement:

```typescript
'use client'

import { useEffect } from 'react'
import { useLandingConfig } from '@/lib/useLandingConfig'

// Client-side redirect to the server's Cognito Managed Login hosted UI in
// sign-up mode. Config comes from a runtime fetch of /landing-config served
// same-origin by the pmcp.run platform (Amplify 200-rewrite -> API GW -> Lambda).
// See cargo-pmcp/CHANGES.md and CR-03 rev-2 for the contract.
export default function Signup() {
  const cfg = useLandingConfig()

  useEffect(() => {
    if (cfg === null || cfg === 'error') return
    if (!cfg.signup) return // rendered as "not available" below

    const { cognitoDomain, cognitoRegion, landingClientId } = cfg.signup
    const origin = window.location.origin
    const url = new URL(
      `https://${cognitoDomain}.auth.${cognitoRegion}.amazoncognito.com/signup`,
    )
    url.searchParams.set('client_id', landingClientId)
    url.searchParams.set('response_type', 'code')
    url.searchParams.set('scope', 'openid email')
    url.searchParams.set('redirect_uri', `${origin}/signup/callback`)
    window.location.replace(url.toString())
  }, [cfg])

  const disabled = cfg === 'error' || (cfg !== null && !cfg.signup)
  return (
    <main className="min-h-screen flex items-center justify-center bg-gray-50">
      <p className="text-gray-600">
        {disabled ? 'Sign-up is not available for this server.' : 'Redirecting to sign-up…'}
      </p>
    </main>
  )
}
```

Deviations from CR-03 §2: added explicit `cfg !== null` in the `disabled` expression to
avoid the implicit-truthiness lint firing on `cfg && !cfg.signup` when `cfg === 'error'`
(the string `'error'` is truthy). Behavior identical to CR-03 intent.

'use client'

import { useEffect } from 'react'
import { useRouter } from 'next/navigation'
import { useLandingConfig } from '@/lib/useLandingConfig'

// Cognito redirects back here with ?code=... after hosted-UI sign-up. We do
// NOT exchange the code for tokens — sign-up alone is the goal for Phase 71.
// The `landing` Cognito client's confidential secret therefore stays server-

export default function SignupCallback() {

  return (
    <main className="min-h-screen flex items-center justify-center bg-gray-50">
      <p className="text-gray-600">Welcome! Redirecting…</p>
    </main>
  )
}

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser -> same-origin `/landing-config` | Platform-served always-200 JSON; Amplify rewrite masks cross-origin boundary from the browser. No CORS, no auth. |
| `cargo new --template landing` scaffold -> end-developer disk | Template files are copied verbatim; no executable code runs at scaffold time. |
| Next.js static export (`output: 'export'`) -> Amplify CDN | No SSR Lambda; static assets only. `NEXT_PUBLIC_*` env vars stripped by platform; previously baked values remain in the legacy bundle until redeploy. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-72.1-01 | Tampering | `/landing-config` response body (MITM on transport) | transfer | Same-origin HTTPS required by the Amplify domain (platform-enforced). Browser enforces TLS. SDK template does not second-guess. |
| T-72.1-02 | Information Disclosure | `landingClientId` rendered in JS bundle | accept | Cognito app client ID is public by design (browser sends it in OAuth authorization code flow). The confidential secret stays server-side (never returned by `/landing-config`). CR-03 §Contract documents this explicitly. |
| T-72.1-03 | Spoofing / Open redirect | `signupRedirectAfter` interpreted as an absolute URL could redirect to attacker site | mitigate | `cargo-pmcp/src/landing/config.rs::validate_relative_path()` (lines 175–200) rejects absolute URLs, protocol-relative URLs, and anything not starting with a single `/`. This validation is pre-existing and unchanged by Phase 72.1 — rustdoc edit only affects the comment text. Template's callback page `router.replace(redirect)` trusts the platform-validated value. |
| T-72.1-04 | Denial of Service | 4 concurrent `/landing-config` fetches on first paint starve the API GW | accept | Response body <200B; Amplify + API GW + Lambda handle this trivially. CR-03 §"Why bundle" + CONTEXT "Hook design" explicitly accept the cost. Module-level cache is a deferred optimization, not a DoS mitigation. |
| T-72.1-05 | Elevation of Privilege | Tampered Cognito domain prefix in `/landing-config` could redirect the user to a spoofed sign-up page | transfer | Attacker would need to compromise DynamoDB / API GW / Lambda — out of the SDK's trust boundary. | T-72.1-06 | Repudiation | Stale cached `/landing-config` serves wrong config after platform update | mitigate | `Cache-Control: no-store` on the response (set by the platform Lambda) + `fetch(..., { cache: 'no-store' })` on the client guarantees fresh reads on every page load. | T-72.1-07 | Information Disclosure | Rustdoc referring to a public env var that no longer exists misleads downstream consumers (SDK-integrators) | mitigate | Task 3 rewrites all three stale `NEXT_PUBLIC_SIGNUP_REDIRECT_AFTER` references. Guardrail G6 enforces 0 matches after the phase. |
