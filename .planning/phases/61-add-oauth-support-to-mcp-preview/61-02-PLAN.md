```javascript
class OAuthManager {
  constructor(runtime) {
    this.runtime = runtime;
    this.codeVerifier = null;
    this.oauthConfig = null;  // Set from /api/config response
    this.popup = null;
    this._boundCallback = (e) => this._handleCallback(e);
    window.addEventListener('message', this._boundCallback);
  }

  /** Store OAuth config from /api/config response */
  setConfig(oauthConfig) {
    this.oauthConfig = oauthConfig;
  }

  /** Whether OAuth is configured for this session */
  get isConfigured() {
    return this.oauthConfig !== null;
  }

  /** Generate a PKCE code_verifier (43-128 char random string, base64url) */
  _generateCodeVerifier() {
    const array = new Uint8Array(32);
    crypto.getRandomValues(array);
    return btoa(String.fromCharCode(...array))
      .replace(/\+/g, '-').replace(/\//g, '_').replace(/=/g, '');
  }

  /** Generate PKCE code_challenge (SHA-256 of verifier, base64url) */
  async _generateCodeChallenge(verifier) {
    const data = new TextEncoder().encode(verifier);
    const hash = await crypto.subtle.digest('SHA-256', data);
    return btoa(String.fromCharCode(...new Uint8Array(hash)))
      .replace(/\+/g, '-').replace(/\//g, '_').replace(/=/g, '');
  }

  /** Start the OAuth login flow. MUST be called from a click handler (popup blocker). */
  async startLogin() {
    if (!this.oauthConfig) return;

    this.codeVerifier = this._generateCodeVerifier();
    const codeChallenge = await this._generateCodeChallenge(this.codeVerifier);
    const state = this._generateCodeVerifier(); // CSRF protection

    const params = new URLSearchParams({
      client_id: this.oauthConfig.client_id,
      response_type: 'code',
      redirect_uri: window.location.origin + '/api/auth/callback',
      scope: this.oauthConfig.scopes.join(' '),
      code_challenge: codeChallenge,
      code_challenge_method: 'S256',
      state: state,
    });

    const authUrl = this.oauthConfig.authorization_endpoint + '?' + params.toString();

    const width = 600, height = 700;
    const left = Math.round((screen.width - width) / 2);
    const top = Math.round((screen.height - height) / 2);
    this.popup = window.open(
      authUrl, 'oauth-popup',
      'width=' + width + ',height=' + height + ',left=' + left + ',top=' + top
    );

    if (!this.popup) {
      this.runtime.addEvent('authError', { error: 'Popup blocked by browser. Allow popups for this site.' });
    }
  }

  /** Handle postMessage from the popup callback page */
  async _handleCallback(event) {
    if (event.origin !== window.location.origin) return;
    if (!event.data || event.data.type !== 'oauth-callback') return;

    const { code, error } = event.data;
    if (this.popup) { this.popup.close(); this.popup = null; }

    if (error) {
      this.runtime.addEvent('authError', { error: 'OAuth error: ' + error });
      this._hideLoginModal();
      return;
    }

    if (!code || !this.codeVerifier) {
      this.runtime.addEvent('authError', { error: 'Missing authorization code or verifier' });
      this._hideLoginModal();
      return;
    }

    // Exchange code for token via the server
    try {
      const resp = await fetch('/api/auth/token-exchange', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          code: code,
          code_verifier: this.codeVerifier,
        }),
      });
      const result = await resp.json();

      if (result.success) {
        this.runtime.addEvent('authSuccess', { message: 'OAuth authentication successful' });
        this._updateAuthStatus(true);
        this._hideLoginModal();
        // Retry session init now that we have a token
        await this.runtime.initSession();
      } else {
        this.runtime.addEvent('authError', { error: result.error || 'Token exchange failed' });
        this._hideLoginModal();
      }
    } catch (e) {
      this.runtime.addEvent('authError', { error: 'Token exchange request failed: ' + e.message });
      this._hideLoginModal();
    }

    this.codeVerifier = null;
  }

  /** Show the login modal overlay */
  showLoginModal() {
    let modal = document.getElementById('oauth-login-modal');
    if (!modal) {
      modal = document.createElement('div');
      modal.id = 'oauth-login-modal';
      modal.innerHTML =
        '<div class="oauth-modal-backdrop">' +
          '<div class="oauth-modal-card">' +
            '<h3>Authentication Required</h3>' +
            '<p>The MCP server requires OAuth authentication.</p>' +
            '<p class="oauth-modal-server">' + (this.runtime.config?.mcp_url || '') + '</p>' +
            '<button id="oauth-sign-in-btn" class="oauth-sign-in-btn">Sign In</button>' +
          '</div>' +
        '</div>';
      document.body.appendChild(modal);
      document.getElementById('oauth-sign-in-btn').addEventListener('click', () => {
        this.startLogin();
      });
    }
    modal.style.display = 'block';
  }

  /** Hide the login modal */
  _hideLoginModal() {
    const modal = document.getElementById('oauth-login-modal');
    if (modal) modal.style.display = 'none';
  }

  /** Show the re-login button in the events panel */
  showReloginPrompt() {
    this.runtime.addEvent('authExpired', {
      error: 'Authentication expired (401/403). Click Re-login to authenticate again.',
      action: 'relogin'
    });
    this._updateAuthStatus(false);
  }

  /** Update the auth status indicator in the header */
  _updateAuthStatus(authenticated) {
    let indicator = document.getElementById('auth-status-indicator');
    if (!indicator && this.isConfigured) {
      // Create the indicator next to the connection status dot
      const headerRight = document.querySelector('.header-status') ||
                          document.querySelector('.connection-status')?.parentElement;
      if (headerRight) {
        indicator = document.createElement('span');
        indicator.id = 'auth-status-indicator';
        indicator.className = 'auth-status';
        headerRight.appendChild(indicator);
      }
    }
    if (indicator) {
      if (authenticated) {
        indicator.textContent = ' Authenticated';
        indicator.className = 'auth-status auth-status-ok';
      } else {
        indicator.innerHTML = ' Auth expired <button class="relogin-btn" id="relogin-btn">Re-login</button>';
        indicator.className = 'auth-status auth-status-expired';
        const btn = document.getElementById('relogin-btn');
        if (btn) btn.addEventListener('click', () => this.showLoginModal());
      }
    }
  }

  destroy() {
    window.removeEventListener('message', this._boundCallback);
  }
}
```

**Update cargo-pmcp/src/commands/preview.rs to construct OAuthPreviewConfig from AuthFlags:**

When `auth_flags.resolve()` returns `AuthMethod::OAuth { ... }`, construct an `OAuthPreviewConfig` and pass it to `PreviewConfig`. The `resolve_auth_header` call should be best-effort -- if CLI OAuth fails (no browser, no cached token), preview should still launch with `oauth_config` set so the browser can handle authentication.

**IMPORTANT: Construct `oauth_config` FIRST (before `resolve_auth_header`), then attempt CLI token as best-effort.** If `resolve_auth_header` fails, log a warning but continue with `auth_header: None` and the browser will handle OAuth via the popup flow.

The key challenge is that `AuthFlags` provides `oauth_issuer` (the OIDC issuer URL) but `OAuthPreviewConfig` needs `authorization_endpoint` and `token_endpoint`. For the browser flow, these are discovered via OIDC discovery (`.well-known/openid-configuration`).

The implementation:

```rust
use crate::commands::flags::AuthMethod;

// ... inside the execute function, after `let auth_method = auth_flags.resolve();` ...

// Build OAuth config for browser-based re-authentication FIRST (if OAuth configured)
// This happens before resolve_auth_header so that even if CLI OAuth fails,
// the browser can still authenticate via the popup flow.
let oauth_config = match &auth_method {
    AuthMethod::OAuth {
        client_id,
        issuer,
        scopes,
        ..
    } => {
        // Try OIDC discovery to find authorization and token endpoints
        match discover_oauth_endpoints(&url, issuer.as_deref()).await {
            Ok((authorization_endpoint, token_endpoint)) => {
                Some(mcp_preview::OAuthPreviewConfig {
                    client_id: client_id.clone(),
                    authorization_endpoint,
                    token_endpoint,
                    scopes: scopes.clone(),
                })
            }
            Err(e) => {
                if global_flags.should_output() {
                    eprintln!(
                        "  {} OAuth browser flow unavailable: {}",
                        "!".yellow(),
                        e
                    );
                    eprintln!(
                        "  {} CLI-acquired token will be used (no browser re-login)",
                        "!".yellow(),
                    );
                }
                None
            }
        }
    }
    _ => None,
};

// Resolve CLI-level auth as best-effort (gets initial token for proxy).
// If this fails but oauth_config is set, the browser handles auth via popup.
let auth_header = match super::auth::resolve_auth_header(&url, &auth_method).await {
    Ok(header) => header,
    Err(e) => {
        if oauth_config.is_some() {
            // Browser will handle OAuth -- CLI failure is non-fatal
            if global_flags.should_output() {
                eprintln!(
                    "  {} CLI token acquisition failed: {} (browser OAuth will handle auth)",
                    "!".yellow(),
                    e
                );
            }
            None
        } else {
            // No browser OAuth fallback -- propagate the error
            return Err(e);
        }
    }
};
```

Add the `discover_oauth_endpoints` helper function at the bottom of preview.rs:

```rust
/// Discover OAuth authorization and token endpoints via OIDC discovery.
///
/// Tries the issuer URL first (if provided), then falls back to the MCP
/// server URL. Fetches `/.well-known/openid-configuration` and extracts
/// `authorization_endpoint` and `token_endpoint`.
async fn discover_oauth_endpoints(
    mcp_url: &str,
    issuer: Option<&str>,
) -> Result<(String, String)> {
    let discovery_urls: Vec<String> = if let Some(iss) = issuer {
        vec![
            format!("{}/.well-known/openid-configuration", iss.trim_end_matches('/')),
        ]
    } else {
        vec![
            format!("{}/.well-known/openid-configuration", mcp_url.trim_end_matches('/')),
        ]
    };

    let client = reqwest::Client::builder()
        .timeout(std::time::Duration::from_secs(10))
        .build()?;

    for url in &discovery_urls {
        match client.get(url).send().await {
            Ok(resp) if resp.status().is_success() => {
                let body: serde_json::Value = resp.json().await?;
                let auth_ep = body
                    .get("authorization_endpoint")
                    .and_then(|v| v.as_str())
                    .map(String::from);
                let token_ep = body
                    .get("token_endpoint")
                    .and_then(|v| v.as_str())
                    .map(String::from);

                if let (Some(auth), Some(token)) = (auth_ep, token_ep) {
                    return Ok((auth, token));
                }

                anyhow::bail!(
                    "OIDC discovery at {} missing authorization_endpoint or token_endpoint",
                    url
                );
            }
            Ok(resp) => {
                tracing::debug!("OIDC discovery at {} returned {}", url, resp.status());
            }
            Err(e) => {
                tracing::debug!("OIDC discovery at {} failed: {}", url, e);
            }
        }
    }

    anyhow::bail!("Could not discover OAuth endpoints (no issuer or MCP server OIDC available)")
}
```
