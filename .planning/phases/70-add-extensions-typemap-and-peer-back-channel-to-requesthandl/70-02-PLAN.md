```rust
pub struct ElicitationManager {
    pending: Arc<RwLock<HashMap<String, oneshot::Sender<ElicitResult>>>>,
    request_tx: Option<mpsc::Sender<ServerRequest>>,
    timeout_duration: Duration,
}

impl ElicitationManager {
    pub async fn elicit_input(&self, request: ElicitRequestParams) -> Result<ElicitResult> {
        let request_tx = self.request_tx.as_ref().ok_or_else(|| {
            Error::protocol(ErrorCode::INTERNAL_ERROR, "Elicitation not configured")
        })?;
        let (tx, rx) = oneshot::channel();
        let elicitation_id = Self::next_elicitation_id();
        self.pending.write().await.insert(elicitation_id.clone(), tx);
        let server_request = ServerRequest::ElicitationCreate(Box::new(request));
        if let Err(e) = request_tx.send(server_request).await {
            self.pending.write().await.remove(&elicitation_id);
            return Err(Error::protocol(ErrorCode::INTERNAL_ERROR, format!("Failed: {e}")));
        }
        match timeout(self.timeout_duration, rx).await {
            Ok(Ok(response)) => Ok(response),
            Ok(Err(_)) => Err(Error::protocol(ErrorCode::INTERNAL_ERROR, "channel closed")),
            Err(_) => {
                self.pending.write().await.remove(&elicitation_id);
                Err(Error::protocol(ErrorCode::REQUEST_TIMEOUT, "timed out"))
            }
        }
    }
}
```

//! Peer back-channel trait for server-to-client RPCs from inside request handlers.
//!
//! Phase 70 / PARITY-HANDLER-01 (HANDLER-05 half).
//!
//! Implementations route outbound RPCs (`sampling/createMessage`, `roots/list`,
//! `notifications/progress`) to the client session that originated the current
//! inbound request. The trait is object-safe so `RequestHandlerExtra` can hold
//! `Option<Arc<dyn PeerHandle>>`.
//!
//! This module is non-wasm only; on wasm32 targets the server dispatch path does
//! not carry a peer, and handlers should treat `extra.peer()` returning `None`
//! as the normal case.

#![cfg(not(target_arch = "wasm32"))]

use crate::error::Result;
use crate::server::roots::ListRootsResult;
use crate::types::progress::ProgressToken;
use crate::types::sampling::{CreateMessageParams, CreateMessageResult};
use async_trait::async_trait;

/// Server-to-client back-channel accessible from inside request handlers.
///
/// # Example
///
/// ```rust,no_run
/// use std::sync::Arc;
/// use pmcp::{RequestHandlerExtra, PeerHandle};
/// # async fn inside_a_tool_handler(extra: RequestHandlerExtra) -> pmcp::Result<()> {
/// if let Some(peer) = extra.peer() {
///     // let result = peer.sample(params).await?;
/// }
/// # Ok(())
/// # }
/// ```
#[async_trait]
pub trait PeerHandle: Send + Sync {
    /// Request the client to sample its LLM (`sampling/createMessage`).
    ///
    /// Routes the request through the same mpsc channel the server uses for
    /// other server-to-client RPCs (ElicitationManager, RootsManager). Honors
    /// the implementation's configured timeout.
    async fn sample(&self, params: CreateMessageParams) -> Result<CreateMessageResult>;

    /// Request the client's root list (`roots/list`).
    async fn list_roots(&self) -> Result<ListRootsResult>;

    /// Send a progress notification (`notifications/progress`).
    ///
    /// Returns `Ok(())` silently when no progress reporter is configured — this
    /// matches the existing `RequestHandlerExtra::report_progress` no-op guard
    /// (see `src/server/cancellation.rs` line ~239).
    async fn progress_notify(
        &self,
        token: ProgressToken,
        progress: f64,
        total: Option<f64>,
        message: Option<String>,
    ) -> Result<()>;
}
```

//! Concrete `PeerHandle` implementation used by the server dispatch path.
//!
//! Phase 70 / PARITY-HANDLER-01. Clone-adapted from `src/server/elicitation.rs`
//! ElicitationManager — same mpsc::Sender<ServerRequest> + HashMap<oneshot::Sender>

#![cfg(not(target_arch = "wasm32"))]

use std::collections::HashMap;
use std::sync::Arc;
use std::time::Duration;

use async_trait::async_trait;
use serde_json::Value;
use tokio::sync::{mpsc, oneshot, RwLock};

use crate::error::{Error, ErrorCode, Result};
use crate::server::progress::ProgressReporter;
use crate::server::roots::ListRootsResult;
use crate::shared::peer::PeerHandle;
use crate::types::progress::ProgressToken;
use crate::types::sampling::{CreateMessageParams, CreateMessageResult};
// NOTE: ServerRequest path — locate via grep during implementation. Likely

use crate::server::ServerRequest;

/// Default timeout for peer.sample() — shorter than ElicitationManager's 5-min default
/// per RESEARCH.md Open Q1 recommendation.
pub(crate) const DEFAULT_PEER_SAMPLE_TIMEOUT: Duration = Duration::from_secs(60);

/// Concrete peer handle constructed per request at every dispatch site.
///
/// Holds a session-scoped mpsc::Sender<ServerRequest> so outbound RPCs route
/// only to the originating client session (mitigates T-70-02 Tampering).
pub(crate) struct DispatchPeerHandle {
    request_tx: mpsc::Sender<ServerRequest>,
    pending: Arc<RwLock<HashMap<String, oneshot::Sender<Value>>>>,
    session_id: Option<String>,
    timeout_duration: Duration,
    progress_reporter: Option<Arc<dyn ProgressReporter>>,
}

impl DispatchPeerHandle {
    pub(crate) fn new(
        request_tx: mpsc::Sender<ServerRequest>,
        session_id: Option<String>,
        timeout_duration: Duration,
    ) -> Self {
        Self {
            request_tx,
            pending: Arc::new(RwLock::new(HashMap::new())),
            session_id,
            timeout_duration,
            progress_reporter: None,
        }
    }

    pub(crate) fn with_progress_reporter(mut self, reporter: Option<Arc<dyn ProgressReporter>>) -> Self {
        self.progress_reporter = reporter;
        self
    }

    fn next_correlation_id() -> String {
        use std::sync::atomic::{AtomicU64, Ordering};
        static COUNTER: AtomicU64 = AtomicU64::new(1);
        format!("peer_{}", COUNTER.fetch_add(1, Ordering::Relaxed))
    }
}

#[async_trait]
impl PeerHandle for DispatchPeerHandle {
    async fn sample(&self, params: CreateMessageParams) -> Result<CreateMessageResult> {
        if self.request_tx.is_closed() {
            return Err(Error::protocol(
                ErrorCode::INTERNAL_ERROR,
                "peer channel closed",
            ));
        }
        let (tx, rx) = oneshot::channel::<Value>();
        let correlation_id = Self::next_correlation_id();
        self.pending.write().await.insert(correlation_id.clone(), tx);

        let server_request = ServerRequest::CreateMessage(Box::new(params));
        if let Err(e) = self.request_tx.send(server_request).await {
            self.pending.write().await.remove(&correlation_id);
            return Err(Error::protocol(
                ErrorCode::INTERNAL_ERROR,
                format!("Failed to send sample request: {e}"),
            ));
        }

        match timeout(self.timeout_duration, rx).await {
            Ok(Ok(value)) => serde_json::from_value::<CreateMessageResult>(value)
                .map_err(|e| Error::protocol(
                    ErrorCode::INTERNAL_ERROR,
                    format!("Invalid sample response: {e}"),
                )),
            Ok(Err(_)) => Err(Error::protocol(
                ErrorCode::INTERNAL_ERROR,
                "sample oneshot channel closed",
            )),
            Err(_) => {
                self.pending.write().await.remove(&correlation_id);
                Err(Error::protocol(
                    ErrorCode::REQUEST_TIMEOUT,
                    "peer.sample timed out",
                ))
            }
        }
    }

    async fn list_roots(&self) -> Result<ListRootsResult> {
        if self.request_tx.is_closed() {
            return Err(Error::protocol(
                ErrorCode::INTERNAL_ERROR,
                "peer channel closed",
            ));
        }
        let (tx, rx) = oneshot::channel::<Value>();
        let correlation_id = Self::next_correlation_id();
        self.pending.write().await.insert(correlation_id.clone(), tx);

        if let Err(e) = self.request_tx.send(ServerRequest::ListRoots).await {
            self.pending.write().await.remove(&correlation_id);
            return Err(Error::protocol(
                ErrorCode::INTERNAL_ERROR,
                format!("Failed to send list_roots request: {e}"),
            ));
        }

        match timeout(self.timeout_duration, rx).await {
            Ok(Ok(value)) => serde_json::from_value::<ListRootsResult>(value)
                .map_err(|e| Error::protocol(
                    ErrorCode::INTERNAL_ERROR,
                    format!("Invalid list_roots response: {e}"),
                )),
            Ok(Err(_)) => Err(Error::protocol(
                ErrorCode::INTERNAL_ERROR,
                "list_roots oneshot channel closed",
            )),
            Err(_) => {
                self.pending.write().await.remove(&correlation_id);
                Err(Error::protocol(
                    ErrorCode::REQUEST_TIMEOUT,
                    "peer.list_roots timed out",
                ))
            }
        }
    }

    async fn progress_notify(
        &self,
        _token: ProgressToken,
        progress: f64,
        total: Option<f64>,
        message: Option<String>,
    ) -> Result<()> {
        if let Some(reporter) = &self.progress_reporter {
            reporter.report_progress(progress, total, message).await
        } else {
            Ok(())
        }
    }

| Boundary | Description |
|----------|-------------|
| handler → client | Untrusted outbound RPC crossing from the server's tool/prompt/resource handler back to the originating client via `peer.sample()` / `peer.list_roots()`. |
| client → handler (response path) | Untrusted CreateMessageResult / ListRootsResult JSON arriving back at `DispatchPeerHandle::sample`/`list_roots` via the pending-oneshot map. |

| server shutdown ↔ in-flight peer call | Peer holds a channel cloned from a server field; a server shutdown or channel drop mid-request must produce a clean error, not a hang. |
