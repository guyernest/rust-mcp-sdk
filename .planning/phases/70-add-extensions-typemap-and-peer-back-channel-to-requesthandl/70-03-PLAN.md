**Step 1 — Create `examples/s42_handler_extensions.rs`:**

```rust
//! Example: using RequestHandlerExtra.extensions typemap for cross-middleware state transfer.
//!
//! Phase 70 / PARITY-HANDLER-01. Demonstrates inserting a typed value into the
//! request-scoped extensions map and retrieving it from inside a tool handler.
//!
//! Run with: cargo run --example s42_handler_extensions

use anyhow::Result;
use async_trait::async_trait;
use pmcp::{RequestHandlerExtra, Server, ServerCapabilities, ToolHandler};
use serde_json::{json, Value};

/// A simple typed value that middleware would inject into extensions
/// before the handler runs. In a real server this would be populated by
/// a middleware layer (e.g., a request-id tagger or user-context loader).
#[derive(Clone, Debug)]
struct RequestContext {
    user_id: u64,
    request_source: String,
}

struct InspectExtensionsTool;

#[async_trait]
impl ToolHandler for InspectExtensionsTool {
    async fn handle(&self, _args: Value, extra: RequestHandlerExtra) -> pmcp::Result<Value> {
        let ctx = extra.extensions().get::<RequestContext>();
        let message = match ctx {
            Some(c) => format!("cross-middleware value retrieved: user_id={}, source={}", c.user_id, c.request_source),
            None => "no RequestContext found in extensions (middleware did not populate it)".to_string(),
        };
        Ok(json!({
            "content": [{"type": "text", "text": message}],
            "isError": false
        }))
    }
}

fn demonstrate_extensions_without_server() {
    // Stand-alone demonstration for `cargo run --example` smoke — builds an
    // in-memory RequestHandlerExtra, simulates the middleware layer
    // inserting a RequestContext, then reads it back.
    let mut extra = RequestHandlerExtra::default();
    extra.extensions_mut().insert(RequestContext {
        user_id: 42,
        request_source: "example".to_string(),
    });
    let ctx = extra.extensions().get::<RequestContext>().expect("value present");
    println!(
        "cross-middleware value retrieved: user_id={}, source={}",
        ctx.user_id, ctx.request_source
    );
}

#[tokio::main]
async fn main() -> Result<()> {
    // For `cargo run --example` smoke: do the in-memory demo + exit,
    // do NOT start a stdio server (that would block the example).
    demonstrate_extensions_without_server();

    // The server construction below is provided as documentation for
    // integration with a real MCP client. It is NOT reached by the
    // example's stdout path because main returns after the demo.
    let _ = Server::builder()
        .name("handler-extensions-example")
        .version("1.0.0")
        .capabilities(ServerCapabilities::tools_only())
        .tool("inspect_extensions", InspectExtensionsTool)
        .build();

    Ok(())
}
```

```rust
//! Example: calling `extra.peer().sample()` from inside a tool handler.
//!
//! Phase 70 / PARITY-HANDLER-01. Contrasts against
//! `examples/s30_tool_with_sampling.rs` which uses the registration-time
//! SamplingHandler trait. This example shows the new ergonomics where a
//! tool handler calls the server-to-client back-channel inline.
//!
//! Run with: cargo run --example s43_handler_peer_sample

#![cfg(not(target_arch = "wasm32"))]

use anyhow::Result;
use async_trait::async_trait;
use pmcp::{RequestHandlerExtra, Server, ServerCapabilities, ToolHandler};
use serde_json::{json, Value};

struct SummarizeViaPeerTool;

#[async_trait]
impl ToolHandler for SummarizeViaPeerTool {
    async fn handle(&self, args: Value, extra: RequestHandlerExtra) -> pmcp::Result<Value> {
        let text = args
            .get("text")
            .and_then(|v| v.as_str())
            .unwrap_or("(no text provided)");

        let message = match extra.peer() {
            Some(_peer) => {
                // In a real server against a connected client:
                //   let result = _peer.sample(CreateMessageParams {
                //       messages: vec![/* ... summarize this: <text> ... */],
                //       ..Default::default()
                //   }).await?;
                //   result.content_text_or_empty()
                //
                // For the example smoke run we don't have a live client,
                // so we document the call shape and return a stub.
                format!(
                    "peer handle available — would call peer.sample() to summarize: {}",
                    text
                )
            }
            None => format!(
                "peer not attached (wasm build, or test fixture without .with_peer) — would summarize inline: {}",
                text
            ),
        };

        Ok(json!({
            "content": [{"type": "text", "text": message}],
            "isError": false
        }))
    }
}

fn demonstrate_peer_not_attached() {
    // A default RequestHandlerExtra has no peer; the example's main path
    // demonstrates the "None branch" of the match above. A full round-trip
    // against a live client is covered by tests/handler_peer_integration.rs.
    let extra = RequestHandlerExtra::default();
    #[cfg(not(target_arch = "wasm32"))]
    let state = if extra.peer().is_some() { "attached" } else { "none" };
    #[cfg(target_arch = "wasm32")]
    let state = "wasm (peer never attached)";
    let _ = extra;
    println!("peer state on a default RequestHandlerExtra: {}", state);
}

#[tokio::main]
async fn main() -> Result<()> {
    demonstrate_peer_not_attached();

    let _ = Server::builder()
        .name("handler-peer-sample-example")
        .version("1.0.0")
        .capabilities(ServerCapabilities::tools_only())
        .tool("summarize_via_peer", SummarizeViaPeerTool)
        .build();

    Ok(())
}
```
