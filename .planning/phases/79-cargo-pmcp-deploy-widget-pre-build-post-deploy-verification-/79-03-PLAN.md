        args: &[&str],
        auth_token: Option<&str>,
        timeout_secs: u64,
        label: &'static str,
    ) -> TestOutcome {
        let exe = match std::env::current_exe() {
            Ok(p) => p,
            Err(e) => return TestOutcome::InfraError(
                InfraErrorKind::Subprocess,
                format!("Failed to resolve current executable: {e}"),
            ),
        };

        cmd.args(args);
        if let Some(token) = auth_token {
            // T-79-04 mitigation — env var, NOT --api-key argv (avoids ps listing leak)

        // per gsd-plan-checker REVISION 1; extended to exit-0 path for N-1

                let _ = s.read_to_string(&mut stdout_buf).await;

    /// F-2 mitigation per gsd-plan-checker REVISION 1.
    /// Tolerant regex parser for `cargo pmcp test conformance` stdout-tail.
    /// Matches "(N/M tests passed)" or "Tests: N/M passed" shapes.
    /// Returns None on unrecognized input -> banner falls back to bare `({duration}ms)`.
    /// Cog target <=12 (single regex + extraction; ≤20 budget per RESEARCH.md table for
    /// new functions added in revision 1).
    pub(crate) fn parse_conformance_summary(stdout: &str) -> Option<TestSummary> {
        // Lazy-static safe via OnceLock; regex compiled once.
        use std::sync::OnceLock;
        static RE: OnceLock<regex::Regex> = OnceLock::new();
        let re = RE.get_or_init(|| {
            // Match either "8/8 tests passed" or "Tests: 8/8 passed".
            regex::Regex::new(r"(?m)(\d+)\s*/\s*(\d+)\s+(?:tests\s+passed|passed)").unwrap()
        });
        let caps = re.captures(stdout)?;
        let passed = caps.get(1)?.as_str().parse::<u32>().ok()?;
        let total = caps.get(2)?.as_str().parse::<u32>().ok()?;
        Some(TestSummary { passed, total })
    }

    /// Extended for N-1/N-3 mitigation per gsd-plan-checker REVISION 2:
    /// recognizes BOTH `widgets failed` (failure path) AND `widgets passed`
    /// (success path) shapes so the success-path banner can emit the verbatim
    /// `(8/8 widgets passed)` annotation.
    /// Tolerant regex parser for `cargo pmcp test apps` stdout-tail.
    /// Matches:
    ///   - "(N/M widgets failed)" -> Some(TestSummary { passed: M-N, total: M }) (failure)
    ///   - "(N/M widgets passed)" -> Some(TestSummary { passed: N, total: M })  (success)
    /// + per-tool `tool 'NAME':` lines (tools list always extracted; empty on success).
    /// Returns (None, vec![]) on unrecognized input -> banner falls back gracefully.
    /// Cog target <=15 (two summary branches + tool extraction).
    pub(crate) fn parse_apps_summary(stdout: &str) -> (Option<TestSummary>, Vec<String>) {
        use std::sync::OnceLock;
        static FAILED_RE: OnceLock<regex::Regex> = OnceLock::new();
        static PASSED_RE: OnceLock<regex::Regex> = OnceLock::new();
        static TOOL_RE: OnceLock<regex::Regex> = OnceLock::new();
        let failed_re = FAILED_RE.get_or_init(|| {
            regex::Regex::new(r"(?m)(\d+)\s*/\s*(\d+)\s+widgets\s+failed").unwrap()
        });
        let passed_re = PASSED_RE.get_or_init(|| {
            regex::Regex::new(r"(?m)(\d+)\s*/\s*(\d+)\s+widgets\s+passed").unwrap()
        });
        let tool_re = TOOL_RE.get_or_init(|| {
            regex::Regex::new(r#"(?m)tool\s+['\"]([^'\"]+)['\"]"#).unwrap()
        });
        // Failure branch takes precedence (defensive; both should never match).
        let summary = failed_re.captures(stdout)
            .and_then(|caps| {
                let failed = caps.get(1)?.as_str().parse::<u32>().ok()?;
                let total = caps.get(2)?.as_str().parse::<u32>().ok()?;
