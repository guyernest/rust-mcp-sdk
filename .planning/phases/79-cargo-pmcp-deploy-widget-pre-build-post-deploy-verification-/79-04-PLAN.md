// Auto-generated by `cargo pmcp app new --embed-widgets`.
// Forces cargo to recompile when widget bundles change.
//
// REVISION 3 HIGH-C1: PMCP_WIDGET_DIRS is a COLON-SEPARATED LIST (Unix PATH
// convention) set by `cargo pmcp deploy` covering all configured [[widgets]] entries.
// REVISION 3 HIGH-G1: when PMCP_WIDGET_DIRS is unset (direct `cargo run` /
// `cargo build` without the deploy wrapper), the template falls back to local
// discovery — walks `CARGO_MANIFEST_DIR + ../widget|widgets/dist` looking for
// the bundle output. Restores the `cargo run` dev loop without requiring
// `cargo pmcp deploy` invocation.
//
// See `cargo pmcp doctor` for a check that this build.rs is present in your crate.
fn main() {
    // CRITICAL: ensures Cargo re-runs build.rs whenever PMCP_WIDGET_DIRS
    // *changes value* between cargo invocations.
    println!("cargo:rerun-if-env-changed=PMCP_WIDGET_DIRS");

    let dirs: Vec<String> = match std::env::var("PMCP_WIDGET_DIRS") {
        Ok(v) if !v.is_empty() => v.split(':').map(String::from).collect(),
        _ => discover_local_widget_dirs(),  // HIGH-G1 fallback
    };

    if dirs.is_empty() {
        println!("cargo:rerun-if-changed=Cargo.toml");
        return;
    }

    for dir in &dirs {
        println!("cargo:rerun-if-changed={dir}");
        if let Ok(entries) = std::fs::read_dir(dir) {
            for entry in entries.flatten() {
                println!("cargo:rerun-if-changed={}", entry.path().display());
            }
        }
    }
}

/// REVISION 3 HIGH-G1: walk up from CARGO_MANIFEST_DIR looking for widget dist dirs.
/// Tries `widget/dist` and `widgets/dist` at the current dir + up to 3 parents.
fn discover_local_widget_dirs() -> Vec<String> {
    let manifest_dir = std::env::var("CARGO_MANIFEST_DIR").unwrap_or_default();
    let candidates = ["widget", "widgets"];
    let mut found = Vec::new();
    for c in &candidates {
        for parent_steps in 0..4 {
            let mut path = std::path::PathBuf::from(&manifest_dir);
            for _ in 0..parent_steps { path.pop(); }
            path.push(c);
            path.push("dist");
            if path.is_dir() {
                found.push(path.to_string_lossy().to_string());
                break;
            }
        }
    }
    found
}

    fn check_widget_rerun_if_changed(quiet: bool) -> u32 {
        // include_str! against widget paths is the failure-mode-B pattern.
        let include_pattern = match Regex::new(r#"include_str!\s*\(\s*"[^"]*widgets?/[^"]*"\s*\)"#) {
            Ok(r) => r,
            Err(_) => return 0,
        };
        let workspace = std::env::current_dir().unwrap_or_else(|_| ".".into());
        // Crates that USE include_str! against widget paths.
        let include_crates = scan_workspace_for_pattern(&workspace, &include_pattern);
        let mut issues = 0u32;
        for krate in include_crates {
            if !crate_has_rerun_if_changed(&krate) {
                if !quiet {
                    println!(

    /// Cog ≤15. Walk workspace src/, filter to .rs files, regex-scan, collect crate dirs.
    fn scan_workspace_for_pattern(workspace: &std::path::Path, pattern: &Regex) -> Vec<std::path::PathBuf> {
        let mut hits: std::collections::BTreeSet<std::path::PathBuf> = Default::default();
        for entry in WalkDir::new(workspace)
            .into_iter()
            .filter_map(Result::ok)
            .filter(|e| e.path().extension().map(|x| x == "rs").unwrap_or(false))
        {
            let path = entry.path();
            if path.components().any(|c| matches!(c.as_os_str().to_str(), Some("target") | Some("node_modules") | Some(".git"))) {
                continue;
            }
            if let Ok(text) = std::fs::read_to_string(path) {
                if pattern.is_match(&text) {
                    if let Some(crate_dir) = find_enclosing_crate_dir(path) {
                        hits.insert(crate_dir);
                    }
                }
            }
        }
        hits.into_iter().collect()
    }

    /// Cog ≤8. Walk up until finding Cargo.toml.
    fn find_enclosing_crate_dir(file: &std::path::Path) -> Option<std::path::PathBuf> {
        let mut p = file.parent()?.to_path_buf();
        loop {
            if p.join("Cargo.toml").is_file() { return Some(p); }
            p = p.parent()?.to_path_buf();
        }
    }

            .map(|s| s.contains("cargo:rerun-if-changed") || s.contains("cargo::rerun-if-changed"))

    }
