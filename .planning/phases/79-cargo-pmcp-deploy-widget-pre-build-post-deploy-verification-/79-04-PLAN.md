    /// Cog ≤18 (Pitfall 6 budget; RESEARCH.md table).
    /// Walks workspace looking for include_str! against widget paths.
    /// For each matched crate, verifies build.rs has cargo:rerun-if-changed.
    /// Returns issue count.
    fn check_widget_rerun_if_changed(quiet: bool) -> u32 {
        let pattern = match Regex::new(r#"include_str!\s*\(\s*"[^"]*widgets?/[^"]*"\s*\)"#) {
            Ok(r) => r,
            Err(_) => return 0,
        };
        let workspace = std::env::current_dir().unwrap_or_else(|_| ".".into());
        let crates = scan_workspace_for_pattern(&workspace, &pattern);
        let mut issues = 0u32;
        for krate in crates {
            if !crate_has_rerun_if_changed(&krate) {
                if !quiet {
                    println!(
                        "  {} crate `{}` includes widget files via include_str! but has no \
                         build.rs cargo:rerun-if-changed; widget changes may not trigger \
                         recompilation.",
                        "!".yellow(),
                        krate.display(),
                    );
                    println!(
                        "    Fix: scaffold a build.rs by running `cargo pmcp app new` to see \
                         the template, OR copy the template documented in cargo-pmcp's README. \
                         Then run `cargo clean -p $(basename '{}')` ONCE so cargo's existing \
                         stale cache doesn't bypass the new build.rs on the first deploy \
                         (Phase 79 Pitfall 1).",
                        krate.display(),
                    );
                }
                issues += 1;
            }
        }
        issues
    }

    /// Cog ≤15. Walk workspace src/, filter to .rs files, regex-scan, collect crate dirs.
    fn scan_workspace_for_pattern(workspace: &std::path::Path, pattern: &Regex) -> Vec<std::path::PathBuf> {
        let mut hits: std::collections::BTreeSet<std::path::PathBuf> = Default::default();
        for entry in WalkDir::new(workspace)
            .into_iter()
            .filter_map(Result::ok)
            .filter(|e| e.path().extension().map(|x| x == "rs").unwrap_or(false))
        {
            let path = entry.path();
            // Skip target/, node_modules/, .git/
            if path.components().any(|c| matches!(c.as_os_str().to_str(), Some("target") | Some("node_modules") | Some(".git"))) {
                continue;
            }
            if let Ok(text) = std::fs::read_to_string(path) {
                if pattern.is_match(&text) {
                    if let Some(crate_dir) = find_enclosing_crate_dir(path) {
                        hits.insert(crate_dir);
                    }
                }
            }
        }
        hits.into_iter().collect()
    }

    /// Cog ≤8. Walk up until finding Cargo.toml.
    fn find_enclosing_crate_dir(file: &std::path::Path) -> Option<std::path::PathBuf> {
        let mut p = file.parent()?.to_path_buf();
        loop {
            if p.join("Cargo.toml").is_file() { return Some(p); }
            p = p.parent()?.to_path_buf();
        }
    }

    /// Cog ≤6.
    fn crate_has_rerun_if_changed(crate_dir: &std::path::Path) -> bool {
        let build_rs = crate_dir.join("build.rs");
        std::fs::read_to_string(&build_rs)
            .map(|s| s.contains("cargo:rerun-if-changed") || s.contains("cargo::rerun-if-changed"))
            .unwrap_or(false)
    }

    //! Example: Phase 79 widget pre-build orchestrator in isolation.
    //!
    //! Demonstrates the build half (Failure Mode A + B mitigation) against a

    //!
    //! For the full end-to-end deploy flow against a real Lambda, use

    //!     cargo run -p cargo-pmcp --example widget_prebuild_demo

        run_widget_build, PackageManager, WidgetConfig,
    };
    use std::path::PathBuf;

    #[tokio::main]
    async fn main() -> anyhow::Result<()> {
        println!("=== Phase 79 — Widget pre-build orchestrator (isolation demo) ===\n");

        // 1. Set up a tempdir with a synthetic widget/.
        let temp = tempfile::tempdir()?;
        let workspace_root = temp.path().to_path_buf();
        let widget_dir = workspace_root.join("widget");
        std::fs::create_dir_all(&widget_dir)?;

        // package-lock.json → npm fallback.

            "version": "0.0.1",

        }"#;

        // node_modules/ pre-created so we skip the install step (avoid network).
        std::fs::create_dir_all(widget_dir.join("node_modules"))?;

        println!("--- 1. Tempdir setup ---");
        println!("  Workspace root: {}", workspace_root.display());
        println!("  Widget dir: {}", widget_dir.display());

        let pm = PackageManager::detect_from_dir(&widget_dir);

            path: "widget".to_string(),
            build: None,
            install: None,
            output_dir: "dist".to_string(),
            embedded_in_crates: vec!["demo-bin".to_string()],

        run_widget_build(&widget, &workspace_root, false).await?;
